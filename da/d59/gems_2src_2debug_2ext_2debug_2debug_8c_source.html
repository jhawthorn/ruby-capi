52static VALUE
53di_body(const rb_debug_inspector_t *dc, void *ptr)
54{
55 VALUE skip_path_prefix = (VALUE)ptr;
56 VALUE locs = rb_debug_inspector_backtrace_locations(dc);
57 VALUE ary = rb_ary_new();
58 long len = RARRAY_LEN(locs);
59 long i;
60
61 for (i=1; i<len; i++) {
62 VALUE loc, e;
63 VALUE iseq = rb_debug_inspector_frame_iseq_get(dc, i);
64
65 if (!NIL_P(iseq)) {
66 VALUE path = iseq_realpath(iseq);
67 if (!NIL_P(path) && !NIL_P(skip_path_prefix) && str_start_with(path, skip_path_prefix)) continue;
68 }
69
70 loc = RARRAY_AREF(locs, i);
71 e = di_entry(loc,
72 rb_debug_inspector_frame_self_get(dc, i),
73 rb_debug_inspector_frame_binding_get(dc, i),
74 iseq,
75 rb_debug_inspector_frame_class_get(dc, i),
76 INT2FIX(len - i));
77 rb_ary_push(ary, e);
78 }
79
80 return ary;
81}
