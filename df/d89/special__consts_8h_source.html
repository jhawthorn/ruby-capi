135static inline bool
136RB_TEST(VALUE obj)
137{
138 /*
139 * Qfalse: ....0000 0000
140 * Qnil: ....0000 1000
141 * ~Qnil: ....1111 0111
142 * v ....xxxx xxxx
143 * ----------------------------
144 * RTEST(v) ....xxxx 0xxx
145 *
146 * RTEST(v) can be 0 if and only if (v == Qfalse || v == Qnil).
147 */
148 return obj & ~RUBY_Qnil;
149}
