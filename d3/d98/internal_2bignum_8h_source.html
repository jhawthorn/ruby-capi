208static inline size_t
209BIGNUM_LEN(VALUE b)
210{
211 if (! BIGNUM_EMBED_P(b)) {
212 return RBIGNUM(b)->as.heap.len;
213 }
214 else {
215 size_t ret = RBASIC(b)->flags;
216 ret &= BIGNUM_EMBED_LEN_MASK;
217 ret >>= BIGNUM_EMBED_LEN_SHIFT;
218 return ret;
219 }
220}
