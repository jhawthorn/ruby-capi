415inline static int
416rb_debug_counter_add(enum rb_debug_counter_type type, int add, int cond)
417{
418 if (cond) {
419 if (ruby_single_main_ractor != NULL) {
420 rb_debug_counter[(int)type] += add;
421 }
422 else {
423 rb_debug_counter_add_atomic(type, add);
424 }
425 }
426 return cond;
427}
428
429inline static int
430rb_debug_counter_max(enum rb_debug_counter_type type, unsigned int num)
431{
432 // TODO: sync
433 if (rb_debug_counter[(int)type] < num) {
434 rb_debug_counter[(int)type] = num;
435 return 1;
436 }
437 else {
438 return 0;
439 }
440}
