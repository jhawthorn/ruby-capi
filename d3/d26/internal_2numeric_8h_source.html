197static inline double
198rb_float_flonum_value(VALUE v)
199{
200#if USE_FLONUM
201 if (v != (VALUE)0x8000000000000002) { /* LIKELY */
202 union {
203 double d;
204 VALUE v;
205 } t;
206
207 VALUE b63 = (v >> 63);
208 /* e: xx1... -> 011... */
209 /* xx0... -> 100... */
210 /* ^b63 */
211 t.v = RUBY_BIT_ROTR((2 - b63) | (v & ~(VALUE)0x03), 3);
212 return t.d;
213 }
214#endif
215 return 0.0;
216}

241static inline VALUE
242rb_float_new_inline(double d)
243{
244#if USE_FLONUM
245 union {
246 double d;
247 VALUE v;
248 } t;
249 int bits;
250
251 t.d = d;
252 bits = (int)((VALUE)(t.v >> 60) & 0x7);
253 /* bits contains 3 bits of b62..b60. */
254 /* bits - 3 = */
255 /* b011 -> b000 */
256 /* b100 -> b001 */
257
258 if (t.v != 0x3000000000000000 /* 1.72723e-77 */ &&
259 !((bits-3) & ~0x01)) {
260 return (RUBY_BIT_ROTL(t.v, 3) & ~(VALUE)0x01) | 0x02;
261 }
262 else if (t.v == (VALUE)0) {
263 /* +0.0 */
264 return 0x8000000000000002;
265 }
266 /* out of range */
267#endif
268 return rb_float_new_in_heap(d);
269}
