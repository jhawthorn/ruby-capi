36static VALUE
37error_pos_str(void)
38{
39 int sourceline;
40 VALUE sourcefile = rb_source_location(&sourceline);
41
42 if (!NIL_P(sourcefile)) {
43 ID caller_name;
44 if (sourceline == 0) {
45 return rb_sprintf("%"PRIsVALUE": ", sourcefile);
46 }
47 else if ((caller_name = rb_frame_callee()) != 0) {
48 return rb_sprintf("%"PRIsVALUE":%d:in `%"PRIsVALUE"': ",
49 sourcefile, sourceline,
50 rb_id2str(caller_name));
51 }
52 else {
53 return rb_sprintf("%"PRIsVALUE":%d: ", sourcefile, sourceline);
54 }
55 }
56 return Qnil;
57}

89static void
90print_errinfo(const VALUE eclass, const VALUE errat, const VALUE emesg, const VALUE str, int highlight)
91{
92 long elen = 0;
93 VALUE mesg;
94
95 if (NIL_P(errat) || RARRAY_LEN(errat) == 0 ||
96 NIL_P(mesg = RARRAY_AREF(errat, 0))) {
97 error_pos(str);
98 }
99 else {
100 write_warn_str(str, mesg);
101 write_warn(str, ": ");
102 }
103
104 if (!NIL_P(emesg)) {
105 elen = RSTRING_LEN(emesg);
106 }
107
108 if (eclass == rb_eRuntimeError && elen == 0) {
109 if (highlight) write_warn(str, underline);
110 write_warn(str, "unhandled exception");
111 if (highlight) write_warn(str, reset);
112 write_warn2(str, "\n", 1);
113 }
114 else {
115 VALUE epath;
116
117 epath = rb_class_name(eclass);
118 if (elen == 0) {
119 if (highlight) write_warn(str, underline);
120 write_warn_str(str, epath);
121 if (highlight) write_warn(str, reset);
122 write_warn(str, "\n");
123 }
124 else {
125 write_warn_str(str, emesg);
126 write_warn(str, "\n");
127 }
128 }

131VALUE
132rb_decorate_message(const VALUE eclass, const VALUE emesg, int highlight)
133{
134 const char *einfo = "";
135 long elen = 0;
136
137 VALUE str = rb_str_new2("");
138
139 if (!NIL_P(emesg)) {
140 einfo = RSTRING_PTR(emesg);
141 elen = RSTRING_LEN(emesg);
142 }
143 if (eclass == rb_eRuntimeError && elen == 0) {
144 if (highlight) write_warn(str, underline);
145 write_warn(str, "unhandled exception");
146 if (highlight) write_warn(str, reset);
147 }
148 else {
149 VALUE epath;
150
151 epath = rb_class_name(eclass);
152 if (elen == 0) {
153 if (highlight) write_warn(str, underline);
154 write_warn_str(str, epath);
155 if (highlight) write_warn(str, reset);
156 }
157 else {
158 /* emesg is a String instance */
159 const char *tail = 0;
160
161 if (highlight) write_warn(str, bold);
162 if (RSTRING_PTR(epath)[0] == '#')
163 epath = 0;
164 if ((tail = memchr(einfo, '\n', elen)) != 0) {
165 write_warn2(str, einfo, tail - einfo);
166 tail++; /* skip newline */
167 }
168 else {
169 write_warn_str(str, emesg);
170 }
171 if (epath) {
172 write_warn(str, " (");
173 if (highlight) write_warn(str, underline);
174 write_warn_str(str, epath);
175 if (highlight) {
176 write_warn(str, reset);
177 write_warn(str, bold);
178 }
179 write_warn2(str, ")", 1);
180 if (highlight) write_warn(str, reset);
181 }
182 if (tail && einfo+elen > tail) {
183 if (!highlight) {
184 write_warn2(str, "\n", 1);
185 write_warn2(str, tail, einfo+elen-tail);
186 }
187 else {
188 elen -= tail - einfo;
189 einfo = tail;
190 write_warn2(str, "\n", 1);
191 while (elen > 0) {
192 tail = memchr(einfo, '\n', elen);
193 if (!tail || tail > einfo) {
194 write_warn(str, bold);
195 write_warn2(str, einfo, tail ? tail-einfo : elen);
196 write_warn(str, reset);
197 if (!tail) {
198 break;
199 }
200 }
201 elen -= tail - einfo;
202 einfo = tail;
203 do ++tail; while (tail < einfo+elen && *tail == '\n');
204 write_warn2(str, einfo, tail-einfo);
205 elen -= tail - einfo;
206 einfo = tail;
207 }
208 }
209 }
210 }
211 }
212
213 return str;
214}

216static void
217print_backtrace(const VALUE eclass, const VALUE errat, const VALUE str, int reverse, long backtrace_limit)
218{
219 if (!NIL_P(errat)) {
220 long i;
221 long len = RARRAY_LEN(errat);
222 const int threshold = 1000000000;
223 int width = (len <= 1) ? INT_MIN : ((int)log10((double)(len > threshold ?
224 ((len - 1) / threshold) :
225 len - 1)) +
226 (len < threshold ? 0 : 9) + 1);
227
228 long skip_start = -1, skip_len = 0;
229
230 // skip for stackoverflow
231 if (eclass == rb_eSysStackError) {
232 long trace_head = 9;
233 long trace_tail = 4;
234 long trace_max = trace_head + trace_tail + 5;
235 if (len > trace_max) {
236 skip_start = trace_head;
237 skip_len = len - trace_max + 5;
238 }
239 }
240
241 // skip for explicit limit
242 if (backtrace_limit >= 0 && len > backtrace_limit + 2) {
243 skip_start = backtrace_limit + 1;
244 skip_len = len - skip_start;
245 }
246
247 for (i = 1; i < len; i++) {
248 if (i == skip_start) {
249 write_warn_str(str, rb_sprintf("\t ... %ld levels...\n", skip_len));
250 i += skip_len;
251 if (i >= len) break;
252 }
253 VALUE line = RARRAY_AREF(errat, reverse ? len - i : i);
254 if (RB_TYPE_P(line, T_STRING)) {
255 VALUE bt = rb_str_new_cstr("\t");
256 if (reverse) rb_str_catf(bt, "%*ld: ", width, len - i);
257 write_warn_str(str, rb_str_catf(bt, "from %"PRIsVALUE"\n", line));
258 }
259 }
260 }

277static void
278show_cause(VALUE errinfo, VALUE str, VALUE opt, VALUE highlight, VALUE reverse, long backtrace_limit, VALUE *shown_causes)
279{
280 VALUE cause = rb_attr_get(errinfo, id_cause);
281 if (!NIL_P(cause) && rb_obj_is_kind_of(cause, rb_eException) &&
282 !shown_cause_p(cause, shown_causes)) {
283 volatile VALUE eclass = CLASS_OF(cause);
284 VALUE errat = rb_get_backtrace(cause);
285 VALUE emesg = rb_get_detailed_message(cause, opt);
286 if (reverse) {
287 show_cause(cause, str, opt, highlight, reverse, backtrace_limit, shown_causes);
288 print_backtrace(eclass, errat, str, TRUE, backtrace_limit);
289 print_errinfo(eclass, errat, emesg, str, RTEST(highlight));
290 }
291 else {
292 print_errinfo(eclass, errat, emesg, str, RTEST(highlight));
293 print_backtrace(eclass, errat, str, FALSE, backtrace_limit);
294 show_cause(cause, str, opt, highlight, reverse, backtrace_limit, shown_causes);

299void
300rb_error_write(VALUE errinfo, VALUE emesg, VALUE errat, VALUE str, VALUE opt, VALUE highlight, VALUE reverse)
301{
302 volatile VALUE eclass;
303 VALUE shown_causes = 0;
304 long backtrace_limit = rb_backtrace_length_limit;
305
306 if (NIL_P(errinfo))
307 return;
308
309 if (errat == Qundef) {
310 errat = Qnil;
311 }
312 eclass = CLASS_OF(errinfo);
313 if (reverse) {
314 static const char traceback[] = "Traceback "
315 "(most recent call last):\n";
316 const int bold_part = rb_strlen_lit("Traceback");
317 char buff[sizeof(traceback)+sizeof(bold)+sizeof(reset)-2], *p = buff;
318 const char *msg = traceback;
319 long len = sizeof(traceback) - 1;
320 if (RTEST(highlight)) {
321#define APPEND(s, l) (memcpy(p, s, l), p += (l))
322 APPEND(bold, sizeof(bold)-1);
323 APPEND(traceback, bold_part);
324 APPEND(reset, sizeof(reset)-1);
325 APPEND(traceback + bold_part, sizeof(traceback)-bold_part-1);
326#undef APPEND
327 len = p - (msg = buff);
328 }
329 write_warn2(str, msg, len);
330 show_cause(errinfo, str, opt, highlight, reverse, backtrace_limit, &shown_causes);
331 print_backtrace(eclass, errat, str, TRUE, backtrace_limit);
332 print_errinfo(eclass, errat, emesg, str, RTEST(highlight));
333 }
334 else {
335 print_errinfo(eclass, errat, emesg, str, RTEST(highlight));
336 print_backtrace(eclass, errat, str, FALSE, backtrace_limit);
337 show_cause(errinfo, str, opt, highlight, reverse, backtrace_limit, &shown_causes);
338 }
339}

341void
342rb_ec_error_print(rb_execution_context_t * volatile ec, volatile VALUE errinfo)
343{
344 volatile uint8_t raised_flag = ec->raised_flag;
345 volatile VALUE errat = Qundef;
346 volatile VALUE emesg = Qundef;
347 volatile bool written = false;
348
349 VALUE opt = rb_hash_new();
350 VALUE highlight = rb_stderr_tty_p() ? Qtrue : Qfalse;
351 rb_hash_aset(opt, ID2SYM(rb_intern_const("highlight")), highlight);
352
353 if (NIL_P(errinfo))
354 return;
355 rb_ec_raised_clear(ec);
356
357 EC_PUSH_TAG(ec);
358 if (EC_EXEC_TAG() == TAG_NONE) {
359 errat = rb_get_backtrace(errinfo);
360 }
361 if (emesg == Qundef) {
362 emesg = Qnil;
363 emesg = rb_get_detailed_message(errinfo, opt);
364 }
365
366 if (!written) {
367 written = true;
368 rb_error_write(errinfo, emesg, errat, Qnil, opt, highlight, Qfalse);
369 }
370
371 EC_POP_TAG();
372 ec->errinfo = errinfo;
373 rb_ec_raised_set(ec, raised_flag);
374}

435static int
436error_handle(rb_execution_context_t *ec, int ex)
437{
438 int status = EXIT_FAILURE;
439
440 if (rb_ec_set_raised(ec))
441 return EXIT_FAILURE;
442 switch (ex & TAG_MASK) {
443 case 0:
444 status = EXIT_SUCCESS;
445 break;
446
447 case TAG_RETURN:
448 error_pos(Qnil);
449 warn_print("unexpected return\n");
450 break;
451 case TAG_NEXT:
452 error_pos(Qnil);
453 warn_print("unexpected next\n");
454 break;
455 case TAG_BREAK:
456 error_pos(Qnil);
457 warn_print("unexpected break\n");
458 break;
459 case TAG_REDO:
460 error_pos(Qnil);
461 warn_print("unexpected redo\n");
462 break;
463 case TAG_RETRY:
464 error_pos(Qnil);
465 warn_print("retry outside of rescue clause\n");
466 break;
467 case TAG_THROW:
468 /* TODO: fix me */
469 error_pos(Qnil);
470 warn_print("unexpected throw\n");
471 break;
472 case TAG_RAISE: {
473 VALUE errinfo = ec->errinfo;
474 if (rb_obj_is_kind_of(errinfo, rb_eSystemExit)) {
475 status = sysexit_status(errinfo);
476 }
477 else if (rb_obj_is_instance_of(errinfo, rb_eSignal) &&
478 rb_ivar_get(errinfo, id_signo) != INT2FIX(SIGSEGV)) {
479 /* no message when exiting by signal */
480 }
481 else if (rb_obj_is_kind_of(errinfo, rb_eSystemCallError) &&
482 FIXNUM_P(rb_attr_get(errinfo, id_signo))) {
483 /* no message when exiting by error to be mapped to signal */
484 }
485 else {
486 rb_ec_error_print(ec, errinfo);
487 }
488 break;
489 }
490 case TAG_FATAL:
491 error_print(ec);
492 break;
493 default:
494 unknown_longjmp_status(ex);
495 break;
496 }
497 rb_ec_reset_raised(ec);
498 return status;
499}
