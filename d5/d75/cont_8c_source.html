326// Allocate some memory from the stack. Used to allocate vm_stack inline with machine stack.
327// @sa fiber_initialize_coroutine
328inline static void *
329fiber_pool_stack_alloca(struct fiber_pool_stack * stack, size_t offset)
330{
331 STACK_GROW_DIR_DETECTION;
332
333 if (DEBUG) fprintf(stderr, "fiber_pool_stack_alloca(%p): %"PRIuSIZE"/%"PRIuSIZE"\n", (void*)stack, offset, stack->available);
334 VM_ASSERT(stack->available >= offset);
335
336 // The pointer to the memory being allocated:
337 void * pointer = STACK_DIR_UPPER(stack->current, (char*)stack->current - offset);
338
339 // Move the stack pointer:
340 stack->current = STACK_DIR_UPPER((char*)stack->current + offset, (char*)stack->current - offset);
341 stack->available -= offset;
342
343 return pointer;
344}

413// Initialize the vacant stack. The [base, size] allocation should not include the guard page.
414// @param base The pointer to the lowest address of the allocated memory.
415// @param size The size of the allocated memory.
416inline static struct fiber_pool_vacancy *
417fiber_pool_vacancy_initialize(struct fiber_pool * fiber_pool, struct fiber_pool_vacancy * vacancies, void * base, size_t size)
418{
419 struct fiber_pool_vacancy * vacancy = fiber_pool_vacancy_pointer(base, size);
420

422 vacancy->stack.size = size;

424 fiber_pool_vacancy_reset(vacancy);

431// Allocate a maximum of count stacks, size given by stride.
432// @param count the number of stacks to allocate / were allocated.
433// @param stride the size of the individual stacks.
434// @return [void *] the allocated memory or NULL if allocation failed.
435inline static void *
436fiber_pool_allocate_memory(size_t * count, size_t stride)
437{
438 // We use a divide-by-2 strategy to try and allocate memory. We are trying
439 // to allocate `count` stacks. In normal situation, this won't fail. But
440 // if we ran out of address space, or we are allocating more memory than
441 // the system would allow (e.g. overcommit * physical memory + swap), we
442 // divide count by two and try again. This condition should only be
443 // encountered in edge cases, but we handle it here gracefully.
444 while (*count > 1) {
445#if defined(_WIN32)
446 void * base = VirtualAlloc(0, (*count)*stride, MEM_COMMIT, PAGE_READWRITE);
447
448 if (!base) {
449 *count = (*count) >> 1;
450 }
451 else {
452 return base;
453 }
454#else
455 errno = 0;
456 void * base = mmap(NULL, (*count)*stride, PROT_READ | PROT_WRITE, FIBER_STACK_FLAGS, -1, 0);
457
458 if (base == MAP_FAILED) {
459 // If the allocation fails, count = count / 2, and try again.
460 *count = (*count) >> 1;
461 }
462 else {
463#if defined(MADV_FREE_REUSE)
464 // On Mac MADV_FREE_REUSE is necessary for the task_info api
465 // to keep the accounting accurate as possible when a page is marked as reusable
466 // it can possibly not occurring at first call thus re-iterating if necessary.
467 while (madvise(base, (*count)*stride, MADV_FREE_REUSE) == -1 && errno == EAGAIN);
468#endif
469 return base;
470 }
471#endif
472 }
473
474 return NULL;
475}

477// Given an existing fiber pool, expand it by the specified number of stacks.
478// @param count the maximum number of stacks to allocate.
479// @return the allocated fiber pool.
480// @sa fiber_pool_allocation_free
481static struct fiber_pool_allocation *
482fiber_pool_expand(struct fiber_pool * fiber_pool, size_t count)
483{
484 STACK_GROW_DIR_DETECTION;
485
486 size_t size = fiber_pool->size;
487 size_t stride = size + RB_PAGE_SIZE;
488
489 // Allocate the memory required for the stacks:
490 void * base = fiber_pool_allocate_memory(&count, stride);
491
492 if (base == NULL) {
493 rb_raise(rb_eFiberError, "can't alloc machine stack to fiber (%"PRIuSIZE" x %"PRIuSIZE" bytes): %s", count, size, ERRNOMSG);
494 }
495
496 struct fiber_pool_vacancy * vacancies = fiber_pool->vacancies;
497 struct fiber_pool_allocation * allocation = RB_ALLOC(struct fiber_pool_allocation);
498
499 // Initialize fiber pool allocation:
500 allocation->base = base;
501 allocation->size = size;
502 allocation->stride = stride;
503 allocation->count = count;
504#ifdef FIBER_POOL_ALLOCATION_FREE
505 allocation->used = 0;
506#endif
507 allocation->pool = fiber_pool;
508
509 if (DEBUG) {
510 fprintf(stderr, "fiber_pool_expand(%"PRIuSIZE"): %p, %"PRIuSIZE"/%"PRIuSIZE" x [%"PRIuSIZE":%"PRIuSIZE"]\n",
511 count, (void*)fiber_pool, fiber_pool->used, fiber_pool->count, size, fiber_pool->vm_stack_size);
512 }
513
514 // Iterate over all stacks, initializing the vacancy list:
515 for (size_t i = 0; i < count; i += 1) {
516 void * base = (char*)allocation->base + (stride * i);
517 void * page = (char*)base + STACK_DIR_UPPER(size, 0);
518
519#if defined(_WIN32)
520 DWORD old_protect;
521
522 if (!VirtualProtect(page, RB_PAGE_SIZE, PAGE_READWRITE | PAGE_GUARD, &old_protect)) {
523 VirtualFree(allocation->base, 0, MEM_RELEASE);
524 rb_raise(rb_eFiberError, "can't set a guard page: %s", ERRNOMSG);
525 }
526#else
527 if (mprotect(page, RB_PAGE_SIZE, PROT_NONE) < 0) {
528 munmap(allocation->base, count*stride);
529 rb_raise(rb_eFiberError, "can't set a guard page: %s", ERRNOMSG);
530 }
531#endif
532
533 vacancies = fiber_pool_vacancy_initialize(
534 fiber_pool, vacancies,
535 (char*)base + STACK_DIR_UPPER(0, RB_PAGE_SIZE),
536 size
537 );
538
539#ifdef FIBER_POOL_ALLOCATION_FREE
540 vacancies->stack.allocation = allocation;
541#endif
542 }
543
544 // Insert the allocation into the head of the pool:
545 allocation->next = fiber_pool->allocations;
546
547#ifdef FIBER_POOL_ALLOCATION_FREE
548 if (allocation->next) {
549 allocation->next->previous = allocation;
550 }
551
552 allocation->previous = NULL;
553#endif
554
555 fiber_pool->allocations = allocation;
556 fiber_pool->vacancies = vacancies;
557 fiber_pool->count += count;
558
559 return allocation;
560}

562// Initialize the specified fiber pool with the given number of stacks.
563// @param vm_stack_size The size of the vm stack to allocate.
564static void
565fiber_pool_initialize(struct fiber_pool * fiber_pool, size_t size, size_t count, size_t vm_stack_size)
566{
567 VM_ASSERT(vm_stack_size < size);
568
569 fiber_pool->allocations = NULL;
570 fiber_pool->vacancies = NULL;
571 fiber_pool->size = ((size / RB_PAGE_SIZE) + 1) * RB_PAGE_SIZE;
572 fiber_pool->count = 0;
573 fiber_pool->initial_count = count;
574 fiber_pool->free_stacks = 1;
575 fiber_pool->used = 0;
576
577 fiber_pool->vm_stack_size = vm_stack_size;
578
579 fiber_pool_expand(fiber_pool, count);
580}

583// Free the list of fiber pool allocations.
584static void
585fiber_pool_allocation_free(struct fiber_pool_allocation * allocation)
586{
587 STACK_GROW_DIR_DETECTION;
588
589 VM_ASSERT(allocation->used == 0);
590
591 if (DEBUG) fprintf(stderr, "fiber_pool_allocation_free: %p base=%p count=%"PRIuSIZE"\n", (void*)allocation, allocation->base, allocation->count);
592
593 size_t i;
594 for (i = 0; i < allocation->count; i += 1) {
595 void * base = (char*)allocation->base + (allocation->stride * i) + STACK_DIR_UPPER(0, RB_PAGE_SIZE);
596
597 struct fiber_pool_vacancy * vacancy = fiber_pool_vacancy_pointer(base, allocation->size);
598
599 // Pop the vacant stack off the free list:
600 fiber_pool_vacancy_remove(vacancy);
601 }
602
603#ifdef _WIN32
604 VirtualFree(allocation->base, 0, MEM_RELEASE);
605#else
606 munmap(allocation->base, allocation->stride * allocation->count);
607#endif
608
609 if (allocation->previous) {
610 allocation->previous->next = allocation->next;
611 }
612 else {
613 // We are the head of the list, so update the pool:
614 allocation->pool->allocations = allocation->next;
615 }
616
617 if (allocation->next) {
618 allocation->next->previous = allocation->previous;
619 }
620
621 allocation->pool->count -= allocation->count;
622
623 ruby_xfree(allocation);
624}
625#endif

627// Acquire a stack from the given fiber pool. If none are available, allocate more.
628static struct fiber_pool_stack
629fiber_pool_stack_acquire(struct fiber_pool * fiber_pool)
630{
631 struct fiber_pool_vacancy * vacancy = fiber_pool_vacancy_pop(fiber_pool);
632
633 if (DEBUG) fprintf(stderr, "fiber_pool_stack_acquire: %p used=%"PRIuSIZE"\n", (void*)fiber_pool->vacancies, fiber_pool->used);
634
635 if (!vacancy) {
636 const size_t maximum = FIBER_POOL_ALLOCATION_MAXIMUM_SIZE;
637 const size_t minimum = fiber_pool->initial_count;
638
639 size_t count = fiber_pool->count;
640 if (count > maximum) count = maximum;
641 if (count < minimum) count = minimum;
642
643 fiber_pool_expand(fiber_pool, count);
644
645 // The free list should now contain some stacks:
646 VM_ASSERT(fiber_pool->vacancies);
647
648 vacancy = fiber_pool_vacancy_pop(fiber_pool);
649 }
650
651 VM_ASSERT(vacancy);
652 VM_ASSERT(vacancy->stack.base);
653
654#if defined(COROUTINE_SANITIZE_ADDRESS)
655 __asan_unpoison_memory_region(fiber_pool_stack_poison_base(&vacancy->stack), fiber_pool_stack_poison_size(&vacancy->stack));
656#endif
657
658 // Take the top item from the free list:
659 fiber_pool->used += 1;
660
661#ifdef FIBER_POOL_ALLOCATION_FREE
662 vacancy->stack.allocation->used += 1;
663#endif
664
665 fiber_pool_stack_reset(&vacancy->stack);
666
667 return vacancy->stack;
668}

670// We advise the operating system that the stack memory pages are no longer being used.
671// This introduce some performance overhead but allows system to relaim memory when there is pressure.
672static inline void
673fiber_pool_stack_free(struct fiber_pool_stack * stack)
674{
675 void * base = fiber_pool_stack_base(stack);
676 size_t size = stack->available;
677
678 // If this is not true, the vacancy information will almost certainly be destroyed:
679 VM_ASSERT(size <= (stack->size - RB_PAGE_SIZE));
680
681 if (DEBUG) fprintf(stderr, "fiber_pool_stack_free: %p+%"PRIuSIZE" [base=%p, size=%"PRIuSIZE"]\n", base, size, stack->base, stack->size);
682
683#if VM_CHECK_MODE > 0 && defined(MADV_DONTNEED)
684 // This immediately discards the pages and the memory is reset to zero.
685 madvise(base, size, MADV_DONTNEED);
686#elif defined(POSIX_MADV_DONTNEED)
687 posix_madvise(base, size, POSIX_MADV_DONTNEED);
688#elif defined(MADV_FREE_REUSABLE)
689 // Acknowledge the kernel down to the task info api we make this
690 // page reusable for future use.
691 // As for MADV_FREE_REUSE below we ensure in the rare occasions the task was not
692 // completed at the time of the call to re-iterate.
693 while (madvise(base, size, MADV_FREE_REUSABLE) == -1 && errno == EAGAIN);
694#elif defined(MADV_FREE)
695 madvise(base, size, MADV_FREE);
696#elif defined(MADV_DONTNEED)
697 madvise(base, size, MADV_DONTNEED);
698#elif defined(_WIN32)
699 VirtualAlloc(base, size, MEM_RESET, PAGE_READWRITE);
700 // Not available in all versions of Windows.
701 //DiscardVirtualMemory(base, size);
702#endif
703
704#if defined(COROUTINE_SANITIZE_ADDRESS)
705 __asan_poison_memory_region(fiber_pool_stack_poison_base(stack), fiber_pool_stack_poison_size(stack));
706#endif
707}

709// Release and return a stack to the vacancy list.
710static void
711fiber_pool_stack_release(struct fiber_pool_stack * stack)
712{
713 struct fiber_pool * pool = stack->pool;
714 struct fiber_pool_vacancy * vacancy = fiber_pool_vacancy_pointer(stack->base, stack->size);
715
716 if (DEBUG) fprintf(stderr, "fiber_pool_stack_release: %p used=%"PRIuSIZE"\n", stack->base, stack->pool->used);
717
718 // Copy the stack details into the vacancy area:
719 vacancy->stack = *stack;
720 // After this point, be careful about updating/using state in stack, since it's copied to the vacancy area.
721
722 // Reset the stack pointers and reserve space for the vacancy data:
723 fiber_pool_vacancy_reset(vacancy);
724
725 // Push the vacancy into the vancancies list:
726 pool->vacancies = fiber_pool_vacancy_push(vacancy, pool->vacancies);
727 pool->used -= 1;
728
729#ifdef FIBER_POOL_ALLOCATION_FREE
730 struct fiber_pool_allocation * allocation = stack->allocation;
731
732 allocation->used -= 1;
733
734 // Release address space and/or dirty memory:
735 if (allocation->used == 0) {
736 fiber_pool_allocation_free(allocation);
737 }
738 else if (stack->pool->free_stacks) {
739 fiber_pool_stack_free(&vacancy->stack);
740 }
741#else
742 // This is entirely optional, but clears the dirty flag from the stack
743 // memory, so it won't get swapped to disk when there is memory pressure:
744 if (stack->pool->free_stacks) {
745 fiber_pool_stack_free(&vacancy->stack);
746 }
747#endif
748}

776static COROUTINE
777fiber_entry(struct coroutine_context * from, struct coroutine_context * to)
778{
779 rb_fiber_t *fiber = to->argument;
780
781#if defined(COROUTINE_SANITIZE_ADDRESS)
782 // Address sanitizer will copy the previous stack base and stack size into
783 // the "from" fiber. `coroutine_initialize_main` doesn't generally know the
784 // stack bounds (base + size). Therefore, the main fiber `stack_base` and
785 // `stack_size` will be NULL/0. It's specifically important in that case to
786 // get the (base+size) of the previous fiber and save it, so that later when
787 // we return to the main coroutine, we don't supply (NULL, 0) to
788 // __sanitizer_start_switch_fiber which royally messes up the internal state
789 // of ASAN and causes (sometimes) the following message:
790 // "WARNING: ASan is ignoring requested __asan_handle_no_return"
791 __sanitizer_finish_switch_fiber(to->fake_stack, (const void**)&from->stack_base, &from->stack_size);
792#endif
793
794 rb_thread_t *thread = fiber->cont.saved_ec.thread_ptr;
795
796#ifdef COROUTINE_PTHREAD_CONTEXT
797 ruby_thread_set_native(thread);
798#endif
799
800 fiber_restore_thread(thread, fiber);
801
802 rb_fiber_start(fiber);
803
804#ifndef COROUTINE_PTHREAD_CONTEXT
805 VM_UNREACHABLE(fiber_entry);
806#endif
807}

809// Initialize a fiber's coroutine's machine stack and vm stack.
810static VALUE *
811fiber_initialize_coroutine(rb_fiber_t *fiber, size_t * vm_stack_size)
812{
813 struct fiber_pool * fiber_pool = fiber->stack.pool;
814 rb_execution_context_t *sec = &fiber->cont.saved_ec;
815 void * vm_stack = NULL;
816
817 VM_ASSERT(fiber_pool != NULL);
818
819 fiber->stack = fiber_pool_stack_acquire(fiber_pool);
820 vm_stack = fiber_pool_stack_alloca(&fiber->stack, fiber_pool->vm_stack_size);
821 *vm_stack_size = fiber_pool->vm_stack_size;
822
823 coroutine_initialize(&fiber->context, fiber_entry, fiber_pool_stack_base(&fiber->stack), fiber->stack.available);
824
825 // The stack for this execution context is the one we allocated:
826 sec->machine.stack_start = fiber->stack.current;
827 sec->machine.stack_maxsize = fiber->stack.available;
828
829 fiber->context.argument = (void*)fiber;
830
831 return vm_stack;
832}

1141cont_save_machine_stack(rb_thread_t *th, rb_context_t *cont)
1142{

1160 cont->machine.stack = ALLOC_N(VALUE, size);
1161 }
1162
1163 FLUSH_REGISTER_WINDOWS;
1164 MEMCPY(cont->machine.stack, cont->machine.stack_src, VALUE, size);
1165}

1270static VALUE
1271cont_capture(volatile int *volatile stat)
1272{
1273 rb_context_t *volatile cont;
1274 rb_thread_t *th = GET_THREAD();
1275 volatile VALUE contval;
1276 const rb_execution_context_t *ec = th->ec;
1277
1278 THREAD_MUST_BE_RUNNING(th);
1279 rb_vm_stack_to_heap(th->ec);
1280 cont = cont_new(rb_cContinuation);
1281 contval = cont->self;
1282
1283#ifdef CAPTURE_JUST_VALID_VM_STACK
1284 cont->saved_vm_stack.slen = ec->cfp->sp - ec->vm_stack;
1285 cont->saved_vm_stack.clen = ec->vm_stack + ec->vm_stack_size - (VALUE*)ec->cfp;
1286 cont->saved_vm_stack.ptr = ALLOC_N(VALUE, cont->saved_vm_stack.slen + cont->saved_vm_stack.clen);

1288 ec->vm_stack,
1289 VALUE, cont->saved_vm_stack.slen);
1290 MEMCPY(cont->saved_vm_stack.ptr + cont->saved_vm_stack.slen,
1291 (VALUE*)ec->cfp,
1292 VALUE,
1293 cont->saved_vm_stack.clen);
1294#else
1295 cont->saved_vm_stack.ptr = ALLOC_N(VALUE, ec->vm_stack_size);
1296 MEMCPY(cont->saved_vm_stack.ptr, ec->vm_stack, VALUE, ec->vm_stack_size);
1297#endif
1298 // At this point, `cfp` is valid but `vm_stack` should be cleared:
1299 rb_ec_set_vm_stack(&cont->saved_ec, NULL, 0);
1300 VM_ASSERT(cont->saved_ec.cfp != NULL);
1301 cont_save_machine_stack(th, cont);
1302
1303 /* backup ensure_list to array for search in another context */
1304 {
1305 rb_ensure_list_t *p;
1306 int size = 0;
1307 rb_ensure_entry_t *entry;
1308 for (p=th->ec->ensure_list; p; p=p->next)
1309 size++;
1310 entry = cont->ensure_array = ALLOC_N(rb_ensure_entry_t,size+1);
1311 for (p=th->ec->ensure_list; p; p=p->next) {
1312 if (!p->entry.marker)
1313 p->entry.marker = rb_ary_tmp_new(0); /* dummy object */
1314 *entry++ = p->entry;
1315 }
1316 entry->marker = 0;
1317 }
1318
1319 if (ruby_setjmp(cont->jmpbuf)) {
1320 VALUE value;
1321
1322 VAR_INITIALIZED(cont);
1323 value = cont->value;
1324 if (cont->argc == -1) rb_exc_raise(value);
1325 cont->value = Qnil;
1326 *stat = 1;
1327 return value;
1328 }
1329 else {
1330 *stat = 0;
1331 return contval;
1332 }
1333}

1335static inline void
1336cont_restore_thread(rb_context_t *cont)
1337{
1338 rb_thread_t *th = GET_THREAD();
1339
1340 /* restore thread context */
1341 if (cont->type == CONTINUATION_CONTEXT) {
1342 /* continuation */
1343 rb_execution_context_t *sec = &cont->saved_ec;
1344 rb_fiber_t *fiber = NULL;
1345
1346 if (sec->fiber_ptr != NULL) {
1347 fiber = sec->fiber_ptr;
1348 }
1349 else if (th->root_fiber) {
1350 fiber = th->root_fiber;
1351 }
1352
1353 if (fiber && th->ec != &fiber->cont.saved_ec) {
1354 ec_switch(th, fiber);
1355 }
1356
1357 if (th->ec->trace_arg != sec->trace_arg) {
1358 rb_raise(rb_eRuntimeError, "can't call across trace_func");
1359 }
1360
1361 /* copy vm stack */
1362#ifdef CAPTURE_JUST_VALID_VM_STACK
1363 MEMCPY(th->ec->vm_stack,
1364 cont->saved_vm_stack.ptr,
1365 VALUE, cont->saved_vm_stack.slen);
1366 MEMCPY(th->ec->vm_stack + th->ec->vm_stack_size - cont->saved_vm_stack.clen,
1367 cont->saved_vm_stack.ptr + cont->saved_vm_stack.slen,
1368 VALUE, cont->saved_vm_stack.clen);
1369#else
1370 MEMCPY(th->ec->vm_stack, cont->saved_vm_stack.ptr, VALUE, sec->vm_stack_size);
1371#endif
1372 /* other members of ec */
1373
1374 th->ec->cfp = sec->cfp;
1375 th->ec->raised_flag = sec->raised_flag;
1376 th->ec->tag = sec->tag;
1377 th->ec->root_lep = sec->root_lep;
1378 th->ec->root_svar = sec->root_svar;
1379 th->ec->ensure_list = sec->ensure_list;
1380 th->ec->errinfo = sec->errinfo;
1381
1382 VM_ASSERT(th->ec->vm_stack != NULL);
1383 }
1384 else {
1385 /* fiber */

1518/*
1519 * Document-class: Continuation
1520 *
1521 * Continuation objects are generated by Kernel#callcc,
1522 * after having +require+d <i>continuation</i>. They hold
1523 * a return address and execution context, allowing a nonlocal return
1524 * to the end of the #callcc block from anywhere within a
1525 * program. Continuations are somewhat analogous to a structured
1526 * version of C's <code>setjmp/longjmp</code> (although they contain
1527 * more state, so you might consider them closer to threads).
1528 *
1529 * For instance:
1530 *
1531 * require "continuation"
1532 * arr = [ "Freddie", "Herbie", "Ron", "Max", "Ringo" ]
1533 * callcc{|cc| $cc = cc}
1534 * puts(message = arr.shift)
1535 * $cc.call unless message =~ /Max/
1536 *
1537 * <em>produces:</em>
1538 *
1539 * Freddie
1540 * Herbie
1541 * Ron
1542 * Max
1543 *
1544 * Also you can call callcc in other methods:
1545 *
1546 * require "continuation"
1547 *
1548 * def g
1549 * arr = [ "Freddie", "Herbie", "Ron", "Max", "Ringo" ]
1550 * cc = callcc { |cc| cc }
1551 * puts arr.shift
1552 * return cc, arr.size
1553 * end
1554 *
1555 * def f
1556 * c, size = g
1557 * c.call(c) if size > 1
1558 * end
1559 *
1560 * f
1561 *
1562 * This (somewhat contrived) example allows the inner loop to abandon
1563 * processing early:
1564 *
1565 * require "continuation"
1566 * callcc {|cont|
1567 * for i in 0..4
1568 * print "#{i}: "
1569 * for j in i*5...(i+1)*5
1570 * cont.call() if j == 17
1571 * printf "%3d", j
1572 * end
1573 * end
1574 * }
1575 * puts
1576 *
1577 * <em>produces:</em>
1578 *
1579 * 0: 0 1 2 3 4
1580 * 1: 5 6 7 8 9
1581 * 2: 10 11 12 13 14
1582 * 3: 15 16
1583 */
1584
1585/*
1586 * call-seq:
1587 * callcc {|cont| block } -> obj
1588 *
1589 * Generates a Continuation object, which it passes to
1590 * the associated block. You need to <code>require
1591 * 'continuation'</code> before using this method. Performing a
1592 * <em>cont</em><code>.call</code> will cause the #callcc
1593 * to return (as will falling through the end of the block). The
1594 * value returned by the #callcc is the value of the
1595 * block, or the value passed to <em>cont</em><code>.call</code>. See
1596 * class Continuation for more details. Also see
1597 * Kernel#throw for an alternative mechanism for
1598 * unwinding a call stack.
1599 */
1600
1601static VALUE
1602rb_callcc(VALUE self)
1603{
1604 volatile int called;
1605 volatile VALUE val = cont_capture(&called);
1606
1607 if (called) {
1608 return val;
1609 }
1610 else {
1611 return rb_yield(val);
1612 }
1613}

1655static inline void
1656rollback_ensure_stack(VALUE self,rb_ensure_list_t *current,rb_ensure_entry_t *target)
1657{
1658 rb_ensure_list_t *p;
1659 rb_ensure_entry_t *entry;
1660 size_t i, j;
1661 size_t cur_size;
1662 size_t target_size;
1663 size_t base_point;
1664 e_proc *func;
1665
1666 cur_size = 0;
1667 for (p=current; p; p=p->next)
1668 cur_size++;
1669 target_size = 0;
1670 for (entry=target; entry->marker; entry++)
1671 target_size++;
1672
1673 /* search common stack point */
1674 p = current;
1675 base_point = cur_size;
1676 while (base_point) {
1677 if (target_size >= base_point &&
1678 p->entry.marker == target[target_size - base_point].marker)
1679 break;
1680 base_point --;
1681 p = p->next;
1682 }
1683
1684 /* rollback function check */
1685 for (i=0; i < target_size - base_point; i++) {
1686 if (!lookup_rollback_func(target[i].e_proc)) {
1687 rb_raise(rb_eRuntimeError, "continuation called from out of critical rb_ensure scope");
1688 }
1689 }
1690 /* pop ensure stack */
1691 while (cur_size > base_point) {
1692 /* escape from ensure block */
1693 (*current->entry.e_proc)(current->entry.data2);
1694 current = current->next;
1695 cur_size--;
1696 }
1697 /* push ensure stack */
1698 for (j = 0; j < i; j++) {
1699 func = lookup_rollback_func(target[i - j - 1].e_proc);
1700 if ((VALUE)func != Qundef) {
1701 (*func)(target[i - j - 1].data2);
1702 }
1703 }
1704}

1708/*
1709 * call-seq:
1710 * cont.call(args, ...)
1711 * cont[args, ...]
1712 *
1713 * Invokes the continuation. The program continues from the end of
1714 * the #callcc block. If no arguments are given, the original #callcc
1715 * returns +nil+. If one argument is given, #callcc returns
1716 * it. Otherwise, an array containing <i>args</i> is returned.
1717 *
1718 * callcc {|cont| cont.call } #=> nil
1719 * callcc {|cont| cont.call 1 } #=> 1
1720 * callcc {|cont| cont.call 1, 2, 3 } #=> [1, 2, 3]
1721 */
1722
1723static VALUE
1724rb_cont_call(int argc, VALUE *argv, VALUE contval)
1725{
1726 rb_context_t *cont = cont_ptr(contval);
1727 rb_thread_t *th = GET_THREAD();
1728
1729 if (cont_thread_value(cont) != th->self) {
1730 rb_raise(rb_eRuntimeError, "continuation called across threads");
1731 }
1732 if (cont->saved_ec.fiber_ptr) {
1733 if (th->ec->fiber_ptr != cont->saved_ec.fiber_ptr) {
1734 rb_raise(rb_eRuntimeError, "continuation called across fiber");
1735 }
1736 }
1737 rollback_ensure_stack(contval, th->ec->ensure_list, cont->ensure_array);
1738
1739 cont->argc = argc;
1740 cont->value = make_passing_arg(argc, argv);
1741
1742 cont_restore_0(cont, &contval);
1743 UNREACHABLE_RETURN(Qnil);
1744}

1750/*
1751 * Document-class: Fiber
1752 *
1753 * Fibers are primitives for implementing light weight cooperative
1754 * concurrency in Ruby. Basically they are a means of creating code blocks
1755 * that can be paused and resumed, much like threads. The main difference
1756 * is that they are never preempted and that the scheduling must be done by
1757 * the programmer and not the VM.
1758 *
1759 * As opposed to other stackless light weight concurrency models, each fiber
1760 * comes with a stack. This enables the fiber to be paused from deeply
1761 * nested function calls within the fiber block. See the ruby(1)
1762 * manpage to configure the size of the fiber stack(s).
1763 *
1764 * When a fiber is created it will not run automatically. Rather it must
1765 * be explicitly asked to run using the Fiber#resume method.
1766 * The code running inside the fiber can give up control by calling
1767 * Fiber.yield in which case it yields control back to caller (the
1768 * caller of the Fiber#resume).
1769 *
1770 * Upon yielding or termination the Fiber returns the value of the last
1771 * executed expression
1772 *
1773 * For instance:
1774 *
1775 * fiber = Fiber.new do
1776 * Fiber.yield 1
1777 * 2
1778 * end
1779 *
1780 * puts fiber.resume
1781 * puts fiber.resume
1782 * puts fiber.resume
1783 *
1784 * <em>produces</em>
1785 *
1786 * 1
1787 * 2
1788 * FiberError: dead fiber called
1789 *
1790 * The Fiber#resume method accepts an arbitrary number of parameters,
1791 * if it is the first call to #resume then they will be passed as
1792 * block arguments. Otherwise they will be the return value of the
1793 * call to Fiber.yield
1794 *
1795 * Example:
1796 *
1797 * fiber = Fiber.new do |first|
1798 * second = Fiber.yield first + 2
1799 * end
1800 *
1801 * puts fiber.resume 10
1802 * puts fiber.resume 1_000_000
1803 * puts fiber.resume "The fiber will be dead before I can cause trouble"
1804 *
1805 * <em>produces</em>
1806 *
1807 * 12
1808 * 1000000
1809 * FiberError: dead fiber called
1810 *
1811 * == Non-blocking Fibers
1812 *
1813 * The concept of <em>non-blocking fiber</em> was introduced in Ruby 3.0.
1814 * A non-blocking fiber, when reaching a operation that would normally block
1815 * the fiber (like <code>sleep</code>, or wait for another process or I/O)
1816 * will yield control to other fibers and allow the <em>scheduler</em> to
1817 * handle blocking and waking up (resuming) this fiber when it can proceed.
1818 *
1819 * For a Fiber to behave as non-blocking, it need to be created in Fiber.new with
1820 * <tt>blocking: false</tt> (which is the default), and Fiber.scheduler
1821 * should be set with Fiber.set_scheduler. If Fiber.scheduler is not set in
1822 * the current thread, blocking and non-blocking fibers' behavior is identical.
1823 *
1824 * Ruby doesn't provide a scheduler class: it is expected to be implemented by
1825 * the user and correspond to Fiber::SchedulerInterface.
1826 *
1827 * There is also Fiber.schedule method, which is expected to immediately perform
1828 * the given block in a non-blocking manner. Its actual implementation is up to
1829 * the scheduler.
1830 *
1831 */
1832

1834 "fiber",
1835 {fiber_mark, fiber_free, fiber_memsize, fiber_compact,},
1836 0, 0, RUBY_TYPED_FREE_IMMEDIATELY
1837};

2286static inline VALUE
2287fiber_switch(rb_fiber_t *fiber, int argc, const VALUE *argv, int kw_splat, rb_fiber_t *resuming_fiber, bool yielding)
2288{
2289 VALUE value;
2290 rb_context_t *cont = &fiber->cont;
2291 rb_thread_t *th = GET_THREAD();
2292
2293 /* make sure the root_fiber object is available */
2294 if (th->root_fiber == NULL) root_fiber_alloc(th);
2295
2296 if (th->ec->fiber_ptr == fiber) {
2297 /* ignore fiber context switch
2298 * because destination fiber is the same as current fiber
2299 */
2300 return make_passing_arg(argc, argv);
2301 }
2302
2303 if (cont_thread_value(cont) != th->self) {
2304 rb_raise(rb_eFiberError, "fiber called across threads");
2305 }
2306
2307 if (FIBER_TERMINATED_P(fiber)) {
2308 value = rb_exc_new2(rb_eFiberError, "dead fiber called");
2309
2310 if (!FIBER_TERMINATED_P(th->ec->fiber_ptr)) {
2311 rb_exc_raise(value);
2312 VM_UNREACHABLE(fiber_switch);
2313 }
2314 else {
2315 /* th->ec->fiber_ptr is also dead => switch to root fiber */
2316 /* (this means we're being called from rb_fiber_terminate, */
2317 /* and the terminated fiber's return_fiber() is already dead) */
2318 VM_ASSERT(FIBER_SUSPENDED_P(th->root_fiber));
2319
2320 cont = &th->root_fiber->cont;
2321 cont->argc = -1;
2322 cont->value = value;
2323
2324 fiber_setcontext(th->root_fiber, th->ec->fiber_ptr);
2325
2326 VM_UNREACHABLE(fiber_switch);
2327 }
2328 }
2329
2330 VM_ASSERT(FIBER_RUNNABLE_P(fiber));
2331
2332 rb_fiber_t *current_fiber = fiber_current();
2333
2334 VM_ASSERT(!current_fiber->resuming_fiber);
2335
2336 if (resuming_fiber) {
2337 current_fiber->resuming_fiber = resuming_fiber;
2338 fiber->prev = fiber_current();
2339 fiber->yielding = 0;
2340 }
2341
2342 VM_ASSERT(!current_fiber->yielding);
2343 if (yielding) {
2344 current_fiber->yielding = 1;
2345 }
2346
2347 if (current_fiber->blocking) {
2348 th->blocking -= 1;
2349 }
2350
2351 cont->argc = argc;
2352 cont->kw_splat = kw_splat;

2354
2355 fiber_store(fiber, th);
2356
2357 // We cannot free the stack until the pthread is joined:
2358#ifndef COROUTINE_PTHREAD_CONTEXT
2359 if (resuming_fiber && FIBER_TERMINATED_P(fiber)) {
2360 fiber_stack_release(fiber);
2361 }
2362#endif
2363
2364 if (fiber_current()->blocking) {

2366 }
2367

2369

2376}

2634/*
2635 * call-seq:
2636 * fiber.transfer(args, ...) -> obj
2637 *
2638 * Transfer control to another fiber, resuming it from where it last
2639 * stopped or starting it if it was not resumed before. The calling
2640 * fiber will be suspended much like in a call to
2641 * Fiber.yield.
2642 *
2643 * The fiber which receives the transfer call treats it much like
2644 * a resume call. Arguments passed to transfer are treated like those
2645 * passed to resume.
2646 *
2647 * The two style of control passing to and from fiber (one is #resume and
2648 * Fiber::yield, another is #transfer to and from fiber) can't be freely
2649 * mixed.
2650 *
2651 * * If the Fiber's lifecycle had started with transfer, it will never
2652 * be able to yield or be resumed control passing, only
2653 * finish or transfer back. (It still can resume other fibers that
2654 * are allowed to be resumed.)
2655 * * If the Fiber's lifecycle had started with resume, it can yield
2656 * or transfer to another Fiber, but can receive control back only
2657 * the way compatible with the way it was given away: if it had
2658 * transferred, it only can be transferred back, and if it had
2659 * yielded, it only can be resumed back. After that, it again can
2660 * transfer or yield.
2661 *
2662 * If those rules are broken FiberError is raised.
2663 *
2664 * For an individual Fiber design, yield/resume is easier to use
2665 * (the Fiber just gives away control, it doesn't need to think
2666 * about who the control is given to), while transfer is more flexible
2667 * for complex cases, allowing to build arbitrary graphs of Fibers
2668 * dependent on each other.
2669 *
2670 *
2671 * Example:
2672 *
2673 * manager = nil # For local var to be visible inside worker block
2674 *
2675 * # This fiber would be started with transfer
2676 * # It can't yield, and can't be resumed
2677 * worker = Fiber.new { |work|
2678 * puts "Worker: starts"
2679 * puts "Worker: Performed #{work.inspect}, transferring back"
2680 * # Fiber.yield # this would raise FiberError: attempt to yield on a not resumed fiber
2681 * # manager.resume # this would raise FiberError: attempt to resume a resumed fiber (double resume)
2682 * manager.transfer(work.capitalize)
2683 * }
2684 *
2685 * # This fiber would be started with resume
2686 * # It can yield or transfer, and can be transferred
2687 * # back or resumed
2688 * manager = Fiber.new {
2689 * puts "Manager: starts"
2690 * puts "Manager: transferring 'something' to worker"
2691 * result = worker.transfer('something')
2692 * puts "Manager: worker returned #{result.inspect}"
2693 * # worker.resume # this would raise FiberError: attempt to resume a transferring fiber
2694 * Fiber.yield # this is OK, the fiber transferred from and to, now it can yield
2695 * puts "Manager: finished"
2696 * }
2697 *
2698 * puts "Starting the manager"
2699 * manager.resume
2700 * puts "Resuming the manager"
2701 * # manager.transfer # this would raise FiberError: attempt to transfer to a yielding fiber
2702 * manager.resume
2703 *
2704 * <em>produces</em>
2705 *
2706 * Starting the manager
2707 * Manager: starts
2708 * Manager: transferring 'something' to worker
2709 * Worker: starts
2710 * Worker: Performed "something", transferring back
2711 * Manager: worker returned "Something"
2712 * Resuming the manager
2713 * Manager: finished
2714 *
2715 */
2716static VALUE

2939/*
2940 * Document-class: Fiber::SchedulerInterface
2941 *
2942 * This is not an existing class, but documentation of the interface that Scheduler
2943 * object should comply to in order to be used as argument to Fiber.scheduler and handle non-blocking
2944 * fibers. See also the "Non-blocking fibers" section in Fiber class docs for explanations
2945 * of some concepts.
2946 *
2947 * Scheduler's behavior and usage are expected to be as follows:
2948 *
2949 * * When the execution in the non-blocking Fiber reaches some blocking operation (like
2950 * sleep, wait for a process, or a non-ready I/O), it calls some of the scheduler's
2951 * hook methods, listed below.
2952 * * Scheduler somehow registers what the current fiber is waiting on, and yields control
2953 * to other fibers with Fiber.yield (so the fiber would be suspended while expecting its
2954 * wait to end, and other fibers in the same thread can perform)
2955 * * At the end of the current thread execution, the scheduler's method #close is called
2956 * * The scheduler runs into a wait loop, checking all the blocked fibers (which it has
2957 * registered on hook calls) and resuming them when the awaited resource is ready
2958 * (e.g. I/O ready or sleep time elapsed).
2959 *
2960 * A typical implementation would probably rely for this closing loop on a gem like
2961 * EventMachine[https://github.com/eventmachine/eventmachine] or
2962 * Async[https://github.com/socketry/async].
2963 *
2964 * This way concurrent execution will be achieved transparently for every
2965 * individual Fiber's code.
2966 *
2967 * Hook methods are:
2968 *
2969 * * #io_wait, #io_read, and #io_write
2970 * * #process_wait
2971 * * #kernel_sleep
2972 * * #timeout_after
2973 * * #address_resolve
2974 * * #block and #unblock
2975 * * (the list is expanded as Ruby developers make more methods having non-blocking calls)
2976 *
2977 * When not specified otherwise, the hook implementations are mandatory: if they are not
2978 * implemented, the methods trying to call hook will fail. To provide backward compatibility,
2979 * in the future hooks will be optional (if they are not implemented, due to the scheduler
2980 * being created for the older Ruby version, the code which needs this hook will not fail,
2981 * and will just behave in a blocking fashion).
2982 *
2983 * It is also strongly recommended that the scheduler implements the #fiber method, which is
2984 * delegated to by Fiber.schedule.
2985 *
2986 * Sample _toy_ implementation of the scheduler can be found in Ruby's code, in
2987 * <tt>test/fiber/scheduler.rb</tt>
2988 *
2989 */
