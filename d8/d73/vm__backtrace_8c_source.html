36inline static int
37calc_pos(const rb_iseq_t *iseq, const VALUE *pc, int *lineno, int *node_id)
38{
39 VM_ASSERT(iseq);
40 VM_ASSERT(ISEQ_BODY(iseq));
41 VM_ASSERT(ISEQ_BODY(iseq)->iseq_encoded);
42 VM_ASSERT(ISEQ_BODY(iseq)->iseq_size);
43 if (! pc) {
44 if (ISEQ_BODY(iseq)->type == ISEQ_TYPE_TOP) {
45 VM_ASSERT(! ISEQ_BODY(iseq)->local_table);
46 VM_ASSERT(! ISEQ_BODY(iseq)->local_table_size);
47 return 0;
48 }
49 if (lineno) *lineno = FIX2INT(ISEQ_BODY(iseq)->location.first_lineno);
50#ifdef USE_ISEQ_NODE_ID
51 if (node_id) *node_id = -1;
52#endif
53 return 1;
54 }
55 else {
56 ptrdiff_t n = pc - ISEQ_BODY(iseq)->iseq_encoded;
57 VM_ASSERT(n <= ISEQ_BODY(iseq)->iseq_size);
58 VM_ASSERT(n >= 0);
59 ASSUME(n >= 0);
60 size_t pos = n; /* no overflow */
61 if (LIKELY(pos)) {
62 /* use pos-1 because PC points next instruction at the beginning of instruction */
63 pos--;
64 }
65#if VMDEBUG && defined(HAVE_BUILTIN___BUILTIN_TRAP)
66 else {
67 /* SDR() is not possible; that causes infinite loop. */
68 rb_print_backtrace();
69 __builtin_trap();
70 }
71#endif
72 if (lineno) *lineno = rb_iseq_line_no(iseq, pos);
73#ifdef USE_ISEQ_NODE_ID
74 if (node_id) *node_id = rb_iseq_node_id(iseq, pos);
75#endif
76 return 1;
77 }
78}

80inline static int
81calc_lineno(const rb_iseq_t *iseq, const VALUE *pc)
82{
83 int lineno;
84 if (calc_pos(iseq, pc, &lineno, NULL)) return lineno;
85 return 0;
86}

401static VALUE
402location_to_str(rb_backtrace_location_t *loc)
403{
404 VALUE file, name;
405 int lineno;
406
407 switch (loc->type) {
408 case LOCATION_TYPE_ISEQ:
409 file = rb_iseq_path(loc->iseq);
410 name = ISEQ_BODY(loc->iseq)->location.label;
411
412 lineno = calc_lineno(loc->iseq, loc->pc);
413 break;
414 case LOCATION_TYPE_CFUNC:
415 if (loc->iseq && loc->pc) {
416 file = rb_iseq_path(loc->iseq);
417 lineno = calc_lineno(loc->iseq, loc->pc);
418 }
419 else {
420 file = GET_VM()->progname;
421 lineno = 0;
422 }
423 name = rb_id2str(loc->mid);
424 break;
425 default:
426 rb_bug("location_to_str: unreachable");
427 }
428
429 return location_format(file, lineno, name);
430}

458static void
459backtrace_mark(void *ptr)
460{
461 rb_backtrace_t *bt = (rb_backtrace_t *)ptr;
462 size_t i, s = bt->backtrace_size;
463
464 for (i=0; i<s; i++) {
465 location_mark_entry(&bt->backtrace[i]);
466 }
467 rb_gc_mark_movable(bt->strary);
468 rb_gc_mark_movable(bt->locary);
469}

536static long
537backtrace_size(const rb_execution_context_t *ec)
538{
539 const rb_control_frame_t *last_cfp = ec->cfp;
540 const rb_control_frame_t *start_cfp = RUBY_VM_END_CONTROL_FRAME(ec);
541
542 if (start_cfp == NULL) {
543 return -1;
544 }
545
546 start_cfp =
547 RUBY_VM_NEXT_CONTROL_FRAME(
548 RUBY_VM_NEXT_CONTROL_FRAME(start_cfp)); /* skip top frames */
549
550 if (start_cfp < last_cfp) {
551 return 0;
552 }
553
554 return start_cfp - last_cfp + 1;
555}

585static VALUE
586rb_ec_partial_backtrace_object(const rb_execution_context_t *ec, long start_frame, long num_frames, int* start_too_large, bool skip_internal, bool do_yield)
587{
588 const rb_control_frame_t *cfp = ec->cfp;
589 const rb_control_frame_t *end_cfp = RUBY_VM_END_CONTROL_FRAME(ec);
590 ptrdiff_t size;
591 rb_backtrace_t *bt;
592 VALUE btobj = backtrace_alloc(rb_cBacktrace);
593 rb_backtrace_location_t *loc = NULL;
594 unsigned long cfunc_counter = 0;
595 GetCoreDataFromValue(btobj, rb_backtrace_t, bt);
596
597 // In the case the thread vm_stack or cfp is not initialized, there is no backtrace.
598 if (end_cfp == NULL) {
599 num_frames = 0;
600 }
601 else {
602 end_cfp = RUBY_VM_NEXT_CONTROL_FRAME(end_cfp);
603
604 /*
605 * top frame (dummy) <- RUBY_VM_END_CONTROL_FRAME
606 * top frame (dummy) <- end_cfp
607 * top frame <- main script
608 * top frame
609 * ...
610 * 2nd frame <- lev:0
611 * current frame <- ec->cfp
612 */
613
614 size = end_cfp - cfp + 1;
615 if (size < 0) {
616 num_frames = 0;
617 }
618 else if (num_frames < 0 || num_frames > size) {
619 num_frames = size;
620 }
621 }
622
623 bt->backtrace = ZALLOC_N(rb_backtrace_location_t, num_frames);
624 bt->backtrace_size = 0;
625 if (num_frames == 0) {
626 if (start_too_large) *start_too_large = 0;
627 return btobj;
628 }
629
630 for (; cfp != end_cfp && (bt->backtrace_size < num_frames); cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(cfp)) {
631 if (cfp->iseq) {
632 if (cfp->pc) {
633 if (start_frame > 0) {
634 start_frame--;
635 }
636 else if (!skip_internal || !is_internal_location(cfp)) {
637 const rb_iseq_t *iseq = cfp->iseq;
638 const VALUE *pc = cfp->pc;
639 loc = &bt->backtrace[bt->backtrace_size++];
640 loc->type = LOCATION_TYPE_ISEQ;
641 loc->iseq = iseq;
642 loc->pc = pc;
643 bt_update_cfunc_loc(cfunc_counter, loc-1, iseq, pc);
644 if (do_yield) {
645 bt_yield_loc(loc - cfunc_counter, cfunc_counter+1, btobj);
646 }
647 cfunc_counter = 0;
648 }
649 }
650 }
651 else {
652 VM_ASSERT(RUBYVM_CFUNC_FRAME_P(cfp));
653 if (start_frame > 0) {
654 start_frame--;
655 }
656 else {
657 loc = &bt->backtrace[bt->backtrace_size++];
658 loc->type = LOCATION_TYPE_CFUNC;
659 loc->iseq = NULL;
660 loc->pc = NULL;
661 loc->mid = rb_vm_frame_method_entry(cfp)->def->original_id;
662 cfunc_counter++;
663 }
664 }
665 }
666
667 if (cfunc_counter > 0) {
668 for (; cfp != end_cfp; cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(cfp)) {
669 if (cfp->iseq && cfp->pc && (!skip_internal || !is_internal_location(cfp))) {
670 bt_update_cfunc_loc(cfunc_counter, loc, cfp->iseq, cfp->pc);
671 if (do_yield) {
672 bt_yield_loc(loc - cfunc_counter, cfunc_counter, btobj);
673 }
674 break;
675 }
676 }
677 }
678
679 if (start_too_large) *start_too_large = (start_frame > 0 ? -1 : 0);
680 return btobj;
681}

689static VALUE
690backtrace_collect(rb_backtrace_t *bt, VALUE (*func)(rb_backtrace_location_t *, void *arg), void *arg)
691{
692 VALUE btary;
693 int i;
694
695 btary = rb_ary_new2(bt->backtrace_size);
696
697 for (i=0; i<bt->backtrace_size; i++) {
698 rb_backtrace_location_t *loc = &bt->backtrace[i];
699 rb_ary_push(btary, func(loc, arg));
700 }
701
702 return btary;
703}

872static void
873backtrace_each(const rb_execution_context_t *ec,
874 void (*init)(void *arg, size_t size),
875 void (*iter_iseq)(void *arg, const rb_control_frame_t *cfp),
876 void (*iter_cfunc)(void *arg, const rb_control_frame_t *cfp, ID mid),
877 void *arg)
878{
879 const rb_control_frame_t *last_cfp = ec->cfp;
880 const rb_control_frame_t *start_cfp = RUBY_VM_END_CONTROL_FRAME(ec);
881 const rb_control_frame_t *cfp;
882 ptrdiff_t size, i;
883
884 // In the case the thread vm_stack or cfp is not initialized, there is no backtrace.
885 if (start_cfp == NULL) {
886 init(arg, 0);
887 return;
888 }
889
890 /* <- start_cfp (end control frame)
891 * top frame (dummy)
892 * top frame (dummy)
893 * top frame <- start_cfp
894 * top frame
895 * ...
896 * 2nd frame <- lev:0
897 * current frame <- ec->cfp
898 */
899
900 start_cfp =
901 RUBY_VM_NEXT_CONTROL_FRAME(
902 RUBY_VM_NEXT_CONTROL_FRAME(start_cfp)); /* skip top frames */
903
904 if (start_cfp < last_cfp) {
905 size = 0;
906 }
907 else {
908 size = start_cfp - last_cfp + 1;
909 }
910
911 init(arg, size);
912
913 /* SDR(); */
914 for (i=0, cfp = start_cfp; i<size; i++, cfp = RUBY_VM_NEXT_CONTROL_FRAME(cfp)) {
915 /* fprintf(stderr, "cfp: %d\n", (rb_control_frame_t *)(ec->vm_stack + ec->vm_stack_size) - cfp); */
916 if (cfp->iseq) {
917 if (cfp->pc) {
918 iter_iseq(arg, cfp);
919 }
920 }
921 else {
922 VM_ASSERT(RUBYVM_CFUNC_FRAME_P(cfp));
923 const rb_callable_method_entry_t *me = rb_vm_frame_method_entry(cfp);
924 ID mid = me->def->original_id;
925
926 iter_cfunc(arg, cfp, mid);
927 }
928 }
929}

1081static VALUE
1082ec_backtrace_to_ary(const rb_execution_context_t *ec, int argc, const VALUE *argv, int lev_default, int lev_plus, int to_str)
1083{
1084 VALUE level, vn;
1085 long lev, n;
1086 VALUE btval;
1087 VALUE r;
1088 int too_large;
1089
1090 rb_scan_args(argc, argv, "02", &level, &vn);
1091
1092 if (argc == 2 && NIL_P(vn)) argc--;
1093
1094 switch (argc) {
1095 case 0:
1096 lev = lev_default + lev_plus;
1097 n = ALL_BACKTRACE_LINES;
1098 break;
1099 case 1:
1100 {
1101 long beg, len, bt_size = backtrace_size(ec);
1102 switch (rb_range_beg_len(level, &beg, &len, bt_size - lev_plus, 0)) {
1103 case Qfalse:
1104 lev = NUM2LONG(level);
1105 if (lev < 0) {
1106 rb_raise(rb_eArgError, "negative level (%ld)", lev);
1107 }
1108 lev += lev_plus;
1109 n = ALL_BACKTRACE_LINES;
1110 break;
1111 case Qnil:
1112 return Qnil;
1113 default:
1114 lev = beg + lev_plus;
1115 n = len;
1116 break;
1117 }
1118 break;
1119 }
1120 case 2:
1121 lev = NUM2LONG(level);
1122 n = NUM2LONG(vn);
1123 if (lev < 0) {
1124 rb_raise(rb_eArgError, "negative level (%ld)", lev);
1125 }
1126 if (n < 0) {
1127 rb_raise(rb_eArgError, "negative size (%ld)", n);
1128 }
1129 lev += lev_plus;
1130 break;
1131 default:
1132 lev = n = 0; /* to avoid warning */
1133 break;
1134 }
1135
1136 if (n == 0) {
1137 return rb_ary_new();
1138 }
1139
1140 btval = rb_ec_partial_backtrace_object(ec, lev, n, &too_large, FALSE, FALSE);
1141
1142 if (too_large) {
1143 return Qnil;
1144 }
1145
1146 if (to_str) {
1147 r = backtrace_to_str_ary(btval);
1148 }
1149 else {
1150 r = backtrace_to_location_ary(btval);
1151 }
1152 RB_GC_GUARD(btval);
1153 return r;
1154}

1548{
1549 int i;
1550 const rb_execution_context_t *ec = GET_EC();

1552 const rb_callable_method_entry_t *cme;
1553
1554 for (i=0; i<limit && cfp != end_cfp;) {
1555 if (VM_FRAME_RUBYFRAME_P(cfp)) {
1556 if (start > 0) {
1557 start--;
1558 continue;
1559 }
1560
1561 /* record frame info */
1562 cme = rb_vm_frame_method_entry(cfp);
1563 if (cme && cme->def->type == VM_METHOD_TYPE_ISEQ) {
1564 buff[i] = (VALUE)cme;
1565 }
1566 else {
1567 buff[i] = (VALUE)cfp->iseq;
1568 }
1569
1570 if (lines) lines[i] = calc_lineno(cfp->iseq, cfp->pc);
1571
1572 i++;
1573 }
1574 else {
1575 cme = rb_vm_frame_method_entry(cfp);
1576 if (cme && cme->def->type == VM_METHOD_TYPE_CFUNC) {
1577 buff[i] = (VALUE)cme;
1578 if (lines) lines[i] = 0;
1579 i++;
1580 }
1581 }
1582 cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(cfp);
1583 }
1584
1585 return i;
