173static void
174clear_method_cache_by_id_in_class(VALUE klass, ID mid)
175{
176 VM_ASSERT(RB_TYPE_P(klass, T_CLASS) || RB_TYPE_P(klass, T_ICLASS));
177 if (rb_objspace_garbage_object_p(klass)) return;
178
179 RB_VM_LOCK_ENTER();
180 if (LIKELY(RCLASS_SUBCLASSES(klass) == NULL)) {
181 // no subclasses
182 // check only current class
183
184 struct rb_id_table *cc_tbl = RCLASS_CC_TBL(klass);
185 VALUE ccs_data;
186
187 // invalidate CCs
188 if (cc_tbl && rb_id_table_lookup(cc_tbl, mid, &ccs_data)) {
189 struct rb_class_cc_entries *ccs = (struct rb_class_cc_entries *)ccs_data;
190 if (NIL_P(ccs->cme->owner)) invalidate_negative_cache(mid);
191 rb_vm_ccs_free(ccs);
192 rb_id_table_delete(cc_tbl, mid);
193 RB_DEBUG_COUNTER_INC(cc_invalidate_leaf_ccs);
194 }
195
196 // remove from callable_m_tbl, if exists
197 struct rb_id_table *cm_tbl;
198 if ((cm_tbl = RCLASS_CALLABLE_M_TBL(klass)) != NULL) {
199 rb_id_table_delete(cm_tbl, mid);
200 RB_DEBUG_COUNTER_INC(cc_invalidate_leaf_callable);
201 }
202 RB_DEBUG_COUNTER_INC(cc_invalidate_leaf);
203 }
204 else {
205 const rb_callable_method_entry_t *cme = complemented_callable_method_entry(klass, mid);
206
207 if (cme) {
208 // invalidate cme if found to invalidate the inline method cache.
209 if (METHOD_ENTRY_CACHED(cme)) {
210 if (METHOD_ENTRY_COMPLEMENTED(cme)) {
211 // do nothing
212 }
213 else {
214 // invalidate cc by invalidating cc->cme
215 VALUE owner = cme->owner;
216 VM_ASSERT(BUILTIN_TYPE(owner) == T_CLASS);
217 VALUE klass_housing_cme;
218 if (cme->def->type == VM_METHOD_TYPE_REFINED && !cme->def->body.refined.orig_me) {
219 klass_housing_cme = owner;
220 }
221 else {
222 klass_housing_cme = RCLASS_ORIGIN(owner);
223 }
224 // replace the cme that will be invalid
225 VM_ASSERT(lookup_method_table(klass_housing_cme, mid) == (const rb_method_entry_t *)cme);
226 const rb_method_entry_t *new_cme = rb_method_entry_clone((const rb_method_entry_t *)cme);
227 rb_method_table_insert(klass_housing_cme, RCLASS_M_TBL(klass_housing_cme), mid, new_cme);
228 }
229
230 vm_cme_invalidate((rb_callable_method_entry_t *)cme);
231 RB_DEBUG_COUNTER_INC(cc_invalidate_tree_cme);
232
233 if (cme->def->iseq_overload) {
234 rb_callable_method_entry_t *monly_cme = (rb_callable_method_entry_t *)lookup_overloaded_cme(cme);
235 if (monly_cme) {
236 vm_cme_invalidate(monly_cme);
237 }
238 }
239 }
240
241 // invalidate complement tbl
242 if (METHOD_ENTRY_COMPLEMENTED(cme)) {
243 VALUE defined_class = cme->defined_class;
244 struct rb_id_table *cm_tbl = RCLASS_CALLABLE_M_TBL(defined_class);
245 VM_ASSERT(cm_tbl != NULL);
246 int r = rb_id_table_delete(cm_tbl, mid);
247 VM_ASSERT(r == TRUE); (void)r;
248 RB_DEBUG_COUNTER_INC(cc_invalidate_tree_callable);
249 }
250
251 RB_DEBUG_COUNTER_INC(cc_invalidate_tree);
252 }
253 else {
254 invalidate_negative_cache(mid);
255 }
256 }
257 RB_VM_LOCK_LEAVE();
258
259 rb_yjit_method_lookup_change(klass, mid);
260}

300static int
301invalidate_all_cc(void *vstart, void *vend, size_t stride, void *data)
302{
303 VALUE v = (VALUE)vstart;
304 for (; v != (VALUE)vend; v += stride) {
305 void *ptr = asan_poisoned_object_p(v);
306 asan_unpoison_object(v, false);
307 if (RBASIC(v)->flags) { // liveness check
308 if (RB_TYPE_P(v, T_CLASS) ||
309 RB_TYPE_P(v, T_ICLASS)) {
310 if (RCLASS_CC_TBL(v)) {
311 rb_cc_table_free(v);
312 }
313 RCLASS_CC_TBL(v) = NULL;
314 }
315 }
316 if (ptr) {
317 asan_poison_object(v);
318 }
319 }
320 return 0; // continue to iteration
321}

396static void
397rb_method_definition_release(rb_method_definition_t *def, int complemented)
398{
399 if (def != NULL) {
400 const int alias_count = def->alias_count;
401 const int complemented_count = def->complemented_count;
402 VM_ASSERT(alias_count >= 0);
403 VM_ASSERT(complemented_count >= 0);
404
405 if (alias_count + complemented_count == 0) {
406 if (METHOD_DEBUG) fprintf(stderr, "-%p-%s:%d,%d (remove)\n", (void *)def,
407 rb_id2name(def->original_id), alias_count, complemented_count);
408 if (def->type == VM_METHOD_TYPE_BMETHOD && def->body.bmethod.hooks) {
409 xfree(def->body.bmethod.hooks);
410 }
411 xfree(def);
412 }
413 else {
414 if (complemented) {
415 VM_ASSERT(def->complemented_count > 0);
416 def->complemented_count--;
417 }
418 else if (def->alias_count > 0) {
419 def->alias_count--;
420 }
421
422 if (METHOD_DEBUG) fprintf(stderr, "-%p-%s:%d->%d,%d->%d (dec)\n", (void *)def, rb_id2name(def->original_id),
423 alias_count, def->alias_count, complemented_count, def->complemented_count);
424 }
425 }

503MJIT_FUNC_EXPORTED void
504rb_method_definition_set(const rb_method_entry_t *me, rb_method_definition_t *def, void *opts)
505{
506 *(rb_method_definition_t **)&me->def = def;
507
508 if (opts != NULL) {
509 switch (def->type) {
510 case VM_METHOD_TYPE_ISEQ:
511 {
512 rb_method_iseq_t *iseq_body = (rb_method_iseq_t *)opts;
513 const rb_iseq_t *iseq = iseq_body->iseqptr;
514 rb_cref_t *method_cref, *cref = iseq_body->cref;
515
516 /* setup iseq first (before invoking GC) */
517 RB_OBJ_WRITE(me, &def->body.iseq.iseqptr, iseq);
518
519 if (ISEQ_BODY(iseq)->mandatory_only_iseq) def->iseq_overload = 1;
520
521 if (0) vm_cref_dump("rb_method_definition_create", cref);
522
523 if (cref) {
524 method_cref = cref;
525 }
526 else {
527 method_cref = vm_cref_new_toplevel(GET_EC()); /* TODO: can we reuse? */
528 }
529
530 RB_OBJ_WRITE(me, &def->body.iseq.cref, method_cref);
531 return;
532 }
533 case VM_METHOD_TYPE_CFUNC:
534 {
535 rb_method_cfunc_t *cfunc = (rb_method_cfunc_t *)opts;
536 setup_method_cfunc_struct(UNALIGNED_MEMBER_PTR(def, body.cfunc), cfunc->func, cfunc->argc);
537 return;
538 }
539 case VM_METHOD_TYPE_ATTRSET:
540 case VM_METHOD_TYPE_IVAR:
541 {
542 const rb_execution_context_t *ec = GET_EC();
543 rb_control_frame_t *cfp;
544 int line;
545
546 def->body.attr.id = (ID)(VALUE)opts;
547
548 cfp = rb_vm_get_ruby_level_next_cfp(ec, ec->cfp);
549
550 if (cfp && (line = rb_vm_get_sourceline(cfp))) {
551 VALUE location = rb_ary_new3(2, rb_iseq_path(cfp->iseq), INT2FIX(line));
552 RB_OBJ_WRITE(me, &def->body.attr.location, rb_ary_freeze(location));
553 }
554 else {
555 VM_ASSERT(def->body.attr.location == 0);
556 }
557 return;
558 }
559 case VM_METHOD_TYPE_BMETHOD:
560 RB_OBJ_WRITE(me, &def->body.bmethod.proc, (VALUE)opts);
561 RB_OBJ_WRITE(me, &def->body.bmethod.defined_ractor, rb_ractor_self(GET_RACTOR()));
562 return;
563 case VM_METHOD_TYPE_NOTIMPLEMENTED:
564 setup_method_cfunc_struct(UNALIGNED_MEMBER_PTR(def, body.cfunc), rb_f_notimplement_internal, -1);
565 return;
566 case VM_METHOD_TYPE_OPTIMIZED:
567 def->body.optimized = *(rb_method_optimized_t *)opts;
568 return;
569 case VM_METHOD_TYPE_REFINED:
570 {
571 const rb_method_refined_t *refined = (rb_method_refined_t *)opts;
572 RB_OBJ_WRITE(me, &def->body.refined.orig_me, refined->orig_me);
573 RB_OBJ_WRITE(me, &def->body.refined.owner, refined->owner);
574 return;
575 }
576 case VM_METHOD_TYPE_ALIAS:
577 RB_OBJ_WRITE(me, &def->body.alias.original_me, (rb_method_entry_t *)opts);
578 return;
579 case VM_METHOD_TYPE_ZSUPER:
580 case VM_METHOD_TYPE_UNDEF:
581 case VM_METHOD_TYPE_MISSING:
582 return;
583 }
584 }

587static void
588method_definition_reset(const rb_method_entry_t *me)
589{
590 rb_method_definition_t *def = me->def;
591
592 switch (def->type) {
593 case VM_METHOD_TYPE_ISEQ:
594 RB_OBJ_WRITTEN(me, Qundef, def->body.iseq.iseqptr);
595 RB_OBJ_WRITTEN(me, Qundef, def->body.iseq.cref);
596 break;
597 case VM_METHOD_TYPE_ATTRSET:
598 case VM_METHOD_TYPE_IVAR:
599 RB_OBJ_WRITTEN(me, Qundef, def->body.attr.location);
600 break;
601 case VM_METHOD_TYPE_BMETHOD:
602 RB_OBJ_WRITTEN(me, Qundef, def->body.bmethod.proc);
603 RB_OBJ_WRITTEN(me, Qundef, def->body.bmethod.defined_ractor);
604 /* give up to check all in a list */
605 if (def->body.bmethod.hooks) rb_gc_writebarrier_remember((VALUE)me);
606 break;
607 case VM_METHOD_TYPE_REFINED:

610 break;
611 case VM_METHOD_TYPE_ALIAS:

613 break;
614 case VM_METHOD_TYPE_CFUNC:
615 case VM_METHOD_TYPE_ZSUPER:
616 case VM_METHOD_TYPE_MISSING:
617 case VM_METHOD_TYPE_OPTIMIZED:
618 case VM_METHOD_TYPE_UNDEF:
619 case VM_METHOD_TYPE_NOTIMPLEMENTED:
620 break;
621 }

698rb_method_entry_complement_defined_class(const rb_method_entry_t *src_me, ID called_id, VALUE defined_class)
699{
700 rb_method_definition_t *def = src_me->def;
701 rb_method_entry_t *me;
702 struct {
703 const struct rb_method_entry_struct *orig_me;
704 VALUE owner;
705 } refined = {0};
706
707 if (!src_me->defined_class &&
708 def->type == VM_METHOD_TYPE_REFINED &&
709 def->body.refined.orig_me) {
710 const rb_method_entry_t *orig_me =
711 rb_method_entry_clone(def->body.refined.orig_me);
712 RB_OBJ_WRITE((VALUE)orig_me, &orig_me->defined_class, defined_class);
713 refined.orig_me = orig_me;
714 refined.owner = orig_me->owner;
715 def = NULL;
716 }
717 else {
718 def = method_definition_addref_complement(def);
719 }
720 me = rb_method_entry_alloc(called_id, src_me->owner, defined_class, def);
721 METHOD_ENTRY_FLAGS_COPY(me, src_me);
722 METHOD_ENTRY_COMPLEMENTED_SET(me);
723 if (!def) {
724 def = rb_method_definition_create(VM_METHOD_TYPE_REFINED, called_id);
725 rb_method_definition_set(me, def, &refined);
726 }
727
728 VM_ASSERT(RB_TYPE_P(me->owner, T_MODULE));
729
730 return (rb_callable_method_entry_t *)me;
731}

825/*
826 * klass->method_table[mid] = method_entry(defined_class, visi, def)
827 *
828 * If def is given (!= NULL), then just use it and ignore original_id and otps.
829 * If not given, then make a new def with original_id and opts.
830 */
831static rb_method_entry_t *
832rb_method_entry_make(VALUE klass, ID mid, VALUE defined_class, rb_method_visibility_t visi,
833 rb_method_type_t type, rb_method_definition_t *def, ID original_id, void *opts)
834{
835 rb_method_entry_t *me;
836 struct rb_id_table *mtbl;
837 st_data_t data;
838 int make_refined = 0;
839 VALUE orig_klass;
840
841 if (NIL_P(klass)) {
842 klass = rb_cObject;
843 }
844 orig_klass = klass;
845
846 if (!FL_TEST(klass, FL_SINGLETON) &&
847 type != VM_METHOD_TYPE_NOTIMPLEMENTED &&
848 type != VM_METHOD_TYPE_ZSUPER) {
849 switch (mid) {
850 case idInitialize:
851 case idInitialize_copy:
852 case idInitialize_clone:
853 case idInitialize_dup:
854 case idRespond_to_missing:
855 visi = METHOD_VISI_PRIVATE;
856 }
857 }
858
859 if (type != VM_METHOD_TYPE_REFINED) {
860 rb_class_modify_check(klass);
861 }
862
863 if (RB_TYPE_P(klass, T_MODULE) && FL_TEST(klass, RMODULE_IS_REFINEMENT)) {
864 VALUE refined_class = rb_refinement_module_get_refined_class(klass);
865 rb_add_refined_method_entry(refined_class, mid);
866 }
867 if (type == VM_METHOD_TYPE_REFINED) {
868 rb_method_entry_t *old_me = lookup_method_table(RCLASS_ORIGIN(klass), mid);
869 if (old_me) rb_vm_check_redefinition_opt_method(old_me, klass);
870 }
871 else {
872 klass = RCLASS_ORIGIN(klass);
873 if (klass != orig_klass) {
874 rb_clear_method_cache(orig_klass, mid);
875 }
876 }
877 mtbl = RCLASS_M_TBL(klass);
878
879 /* check re-definition */
880 if (rb_id_table_lookup(mtbl, mid, &data)) {
881 rb_method_entry_t *old_me = (rb_method_entry_t *)data;
882 rb_method_definition_t *old_def = old_me->def;
883
884 if (rb_method_definition_eq(old_def, def)) return old_me;
885 rb_vm_check_redefinition_opt_method(old_me, klass);
886
887 if (old_def->type == VM_METHOD_TYPE_REFINED) make_refined = 1;
888
889 if (RTEST(ruby_verbose) &&
890 type != VM_METHOD_TYPE_UNDEF &&
891 (old_def->alias_count == 0) &&
892 (!old_def->no_redef_warning) &&
893 !make_refined &&
894 old_def->type != VM_METHOD_TYPE_UNDEF &&
895 old_def->type != VM_METHOD_TYPE_ZSUPER &&
896 old_def->type != VM_METHOD_TYPE_ALIAS) {
897 const rb_iseq_t *iseq = 0;
898
899 rb_warning("method redefined; discarding old %"PRIsVALUE, rb_id2str(mid));
900 switch (old_def->type) {
901 case VM_METHOD_TYPE_ISEQ:
902 iseq = def_iseq_ptr(old_def);
903 break;
904 case VM_METHOD_TYPE_BMETHOD:
905 iseq = rb_proc_get_iseq(old_def->body.bmethod.proc, 0);
906 break;
907 default:
908 break;
909 }
910 if (iseq) {
911 rb_compile_warning(RSTRING_PTR(rb_iseq_path(iseq)),
912 FIX2INT(ISEQ_BODY(iseq)->location.first_lineno),
913 "previous definition of %"PRIsVALUE" was here",
914 rb_id2str(old_def->original_id));
915 }
916 }
917 }
918
919 /* create method entry */
920 me = rb_method_entry_create(mid, defined_class, visi, NULL);
921 if (def == NULL) def = rb_method_definition_create(type, original_id);
922 rb_method_definition_set(me, def, opts);
923
924 rb_clear_method_cache(klass, mid);
925
926 /* check mid */
927 if (klass == rb_cObject) {
928 switch (mid) {
929 case idInitialize:
930 case idRespond_to_missing:
931 case idMethodMissing:
932 case idRespond_to:
933 rb_warn("redefining Object#%s may cause infinite loop", rb_id2name(mid));
934 }
935 }
936 /* check mid */
937 if (mid == object_id || mid == id__send__) {
938 if (type == VM_METHOD_TYPE_ISEQ && search_method(klass, mid, 0)) {
939 rb_warn("redefining `%s' may cause serious problems", rb_id2name(mid));
940 }
941 }
942
943 if (make_refined) {
944 make_method_entry_refined(klass, me);
945 }
946
947 rb_method_table_insert(klass, mtbl, mid, me);
948
949 VM_ASSERT(me->def != NULL);
950
951 /* check optimized method override by a prepended module */
952 if (RB_TYPE_P(orig_klass, T_MODULE)) {
953 check_override_opt_method(klass, (VALUE)mid);
954 }
955
956 return me;
957}

1225static inline const rb_callable_method_entry_t *
1226prepare_callable_method_entry(VALUE defined_class, ID id, const rb_method_entry_t * const me, int create)

1243 VM_ASSERT(callable_method_entry_p(cme));

1249 }
1250 cme = rb_method_entry_complement_defined_class(me, me->called_id, defined_class);

1252 RB_OBJ_WRITTEN(defined_class, Qundef, (VALUE)cme);
1253 VM_ASSERT(callable_method_entry_p(cme));
1254 }
1255 else {
1256 return NULL;
1257 }
1258 }
1259 else {
1260 cme = (const rb_callable_method_entry_t *)me;
1261 VM_ASSERT(callable_method_entry_p(cme));
1262 VM_ASSERT(!METHOD_ENTRY_INVALIDATED(cme));
1263 }

1265 }

1466static const rb_method_entry_t *
1467resolve_refined_method(VALUE refinements, const rb_method_entry_t *me, VALUE *defined_class_ptr)
1468{
1469 while (me && me->def->type == VM_METHOD_TYPE_REFINED) {

1472 VALUE super;
1473
1474 refinement = find_refinement(refinements, me->owner);
1475 if (!NIL_P(refinement)) {

1477
1478 if (tmp_me && tmp_me->def->type != VM_METHOD_TYPE_REFINED) {
1479 return tmp_me;

1486 return tmp_me;
1487 }

1489 super = RCLASS_SUPER(me->owner);
1490 if (!super) {
1491 return 0;
1492 }
1493
1494 me = search_method_protect(super, me->called_id, defined_class_ptr);
1495 }
1496 return me;

1522{
1523 VALUE data;
1524 rb_method_entry_t *me = 0;
1525 VALUE self = klass;
1526
1527 klass = RCLASS_ORIGIN(klass);

1531 }

1778/*
1779 * call-seq:
1780 * undef_method(symbol) -> self
1781 * undef_method(string) -> self
1782 *
1783 * Prevents the current class from responding to calls to the named
1784 * method. Contrast this with <code>remove_method</code>, which deletes
1785 * the method from the particular class; Ruby will still search
1786 * superclasses and mixed-in modules for a possible receiver.
1787 * String arguments are converted to symbols.
1788 *
1789 * class Parent
1790 * def hello
1791 * puts "In parent"
1792 * end
1793 * end
1794 * class Child < Parent
1795 * def hello
1796 * puts "In child"
1797 * end
1798 * end
1799 *
1800 *
1801 * c = Child.new
1802 * c.hello
1803 *
1804 *
1805 * class Child
1806 * remove_method :hello # remove from child, still in parent
1807 * end
1808 * c.hello
1809 *
1810 *
1811 * class Child
1812 * undef_method :hello # prevent any calls to 'hello'
1813 * end
1814 * c.hello
1815 *
1816 * <em>produces:</em>
1817 *
1818 * In child
1819 * In parent
1820 * prog.rb:23: undefined method `hello' for #<Child:0x401b3bb4> (NoMethodError)
1821 */
1822
1823static VALUE
1824rb_mod_undef_method(int argc, VALUE *argv, VALUE mod)
1825{
1826 int i;
1827 for (i = 0; i < argc; i++) {
1828 VALUE v = argv[i];
1829 ID id = rb_check_id(&v);
1830 if (!id) {
1831 rb_method_name_error(mod, v);
1832 }
1833 rb_undef(mod, id);
1834 }
1835 return mod;
1836}

2372/*
2373 * call-seq:
2374 * ruby2_keywords(method_name, ...) -> nil
2375 *
2376 * For the given method names, marks the method as passing keywords through
2377 * a normal argument splat. This should only be called on methods that
2378 * accept an argument splat (<tt>*args</tt>) but not explicit keywords or
2379 * a keyword splat. It marks the method such that if the method is called
2380 * with keyword arguments, the final hash argument is marked with a special
2381 * flag such that if it is the final element of a normal argument splat to
2382 * another method call, and that method call does not include explicit
2383 * keywords or a keyword splat, the final element is interpreted as keywords.
2384 * In other words, keywords will be passed through the method to other
2385 * methods.
2386 *
2387 * This should only be used for methods that delegate keywords to another
2388 * method, and only for backwards compatibility with Ruby versions before 3.0.
2389 * See https://www.ruby-lang.org/en/news/2019/12/12/separation-of-positional-and-keyword-arguments-in-ruby-3-0/
2390 * for details on why +ruby2_keywords+ exists and when and how to use it.
2391 *
2392 * This method will probably be removed at some point, as it exists only
2393 * for backwards compatibility. As it does not exist in Ruby versions before
2394 * 2.7, check that the module responds to this method before calling it:
2395 *
2396 * module Mod
2397 * def foo(meth, *args, &block)
2398 * send(:"do_#{meth}", *args, &block)
2399 * end
2400 * ruby2_keywords(:foo) if respond_to?(:ruby2_keywords, true)
2401 * end
2402 *
2403 * However, be aware that if the +ruby2_keywords+ method is removed, the
2404 * behavior of the +foo+ method using the above approach will change so that
2405 * the method does not pass through keywords.
2406 */

2410{
2411 int i;
2412 VALUE origin_class = RCLASS_ORIGIN(module);
2413
2414 rb_check_arity(argc, 1, UNLIMITED_ARGUMENTS);
2415 rb_check_frozen(module);
2416
2417 for (i = 0; i < argc; i++) {
2418 VALUE v = argv[i];
2419 ID name = rb_check_id(&v);
2420 rb_method_entry_t *me;
2421 VALUE defined_class;
2422
2423 if (!name) {
2424 rb_print_undef_str(module, v);

2426

2428 if (!me && RB_TYPE_P(module, T_MODULE)) {
2429 me = search_method(rb_cObject, name, &defined_class);
2430 }
2431
2432 if (UNDEFINED_METHOD_ENTRY_P(me) ||
2433 UNDEFINED_REFINED_METHOD_P(me->def)) {
2434 rb_print_undef(module, name, METHOD_VISI_UNDEF);
2435 }
2436
2437 if (module == defined_class || origin_class == defined_class) {
2438 switch (me->def->type) {
2439 case VM_METHOD_TYPE_ISEQ:

2441 !ISEQ_BODY(me->def->body.iseq.iseqptr)->param.flags.has_kw &&
2442 !ISEQ_BODY(me->def->body.iseq.iseqptr)->param.flags.has_kwrest) {

2464 }

2467 }
2468 break;
2469 }
2470 }
2471 /* fallthrough */
2472 default:

2476 }
2477 else {

2479 }
