78static inline const rb_control_frame_t *
79rb_vm_search_cf_from_ep(const rb_execution_context_t *ec, const rb_control_frame_t *cfp, const VALUE * const ep)
80{
81 if (!ep) {
82 return NULL;
83 }
84 else {
85 const rb_control_frame_t * const eocfp = RUBY_VM_END_CONTROL_FRAME(ec); /* end of control frame pointer */
86
87 while (cfp < eocfp) {
88 if (cfp->ep == ep) {
89 return cfp;
90 }
91 cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(cfp);
92 }
93
94 return NULL;
95 }

229static rb_cref_t *
230vm_cref_new0(VALUE klass, rb_method_visibility_t visi, int module_func, rb_cref_t *prev_cref, int pushed_by_eval, int use_prev_prev, int singleton)
231{
232 VALUE refinements = Qnil;
233 int omod_shared = FALSE;
234 rb_cref_t *cref;
235
236 /* scope */
237 union {
238 rb_scope_visibility_t visi;
239 VALUE value;
240 } scope_visi;
241
242 scope_visi.visi.method_visi = visi;
243 scope_visi.visi.module_func = module_func;
244
245 /* refinements */
246 if (prev_cref != NULL && prev_cref != (void *)1 /* TODO: why CREF_NEXT(cref) is 1? */) {
247 refinements = CREF_REFINEMENTS(prev_cref);
248
249 if (!NIL_P(refinements)) {
250 omod_shared = TRUE;
251 CREF_OMOD_SHARED_SET(prev_cref);
252 }
253 }
254
255 VM_ASSERT(singleton || klass);
256
257 cref = (rb_cref_t *)rb_imemo_new(imemo_cref, klass, (VALUE)(use_prev_prev ? CREF_NEXT(prev_cref) : prev_cref), scope_visi.value, refinements);
258
259 if (pushed_by_eval) CREF_PUSHED_BY_EVAL_SET(cref);
260 if (omod_shared) CREF_OMOD_SHARED_SET(cref);
261 if (singleton) CREF_SINGLETON_SET(cref);
262
263 return cref;
264}

284static rb_cref_t *
285vm_cref_dup(const rb_cref_t *cref)
286{
287 const rb_scope_visibility_t *visi = CREF_SCOPE_VISI(cref);
288 rb_cref_t *next_cref = CREF_NEXT(cref), *new_cref;
289 int pushed_by_eval = CREF_PUSHED_BY_EVAL(cref);
290 int singleton = CREF_SINGLETON(cref);
291
292 new_cref = vm_cref_new(cref->klass_or_self, visi->method_visi, visi->module_func, next_cref, pushed_by_eval, singleton);
293
294 if (!NIL_P(CREF_REFINEMENTS(cref))) {
295 VALUE ref = rb_hash_dup(CREF_REFINEMENTS(cref));
296 rb_hash_foreach(ref, ref_delete_symkey, Qnil);
297 CREF_REFINEMENTS_SET(new_cref, ref);
298 CREF_OMOD_SHARED_UNSET(new_cref);
299 }
300
301 return new_cref;
302}

468MJIT_FUNC_EXPORTED int
469rb_dtrace_setup(rb_execution_context_t *ec, VALUE klass, ID id,
470 struct ruby_dtrace_method_hook_args *args)
471{
472 enum ruby_value_type type;
473 if (!klass) {
474 if (!ec) ec = GET_EC();
475 if (!rb_ec_frame_method_id_and_class(ec, &id, 0, &klass) || !klass)
476 return FALSE;
477 }
478 if (RB_TYPE_P(klass, T_ICLASS)) {
479 klass = RBASIC(klass)->klass;
480 }
481 else if (FL_TEST(klass, FL_SINGLETON)) {
482 klass = rb_attr_get(klass, id__attached__);
483 if (NIL_P(klass)) return FALSE;
484 }
485 type = BUILTIN_TYPE(klass);
486 if (type == T_CLASS || type == T_ICLASS || type == T_MODULE) {
487 VALUE name = rb_class_path(klass);
488 const char *classname, *filename;
489 const char *methodname = rb_id2name(id);
490 if (methodname && (filename = rb_source_location_cstr(&args->line_no)) != 0) {
491 if (NIL_P(name) || !(classname = StringValuePtr(name)))
492 classname = "<unknown>";
493 args->classname = classname;
494 args->methodname = methodname;
495 args->filename = filename;
496 args->klass = klass;
497 args->name = name;
498 return TRUE;
499 }
500 }
501 return FALSE;
502}

504/*
505 * call-seq:
506 * RubyVM.stat -> Hash
507 * RubyVM.stat(hsh) -> hsh
508 * RubyVM.stat(Symbol) -> Numeric
509 *
510 * Returns a Hash containing implementation-dependent counters inside the VM.
511 *
512 * This hash includes information about method/constant caches:
513 *
514 * {
515 * :constant_cache_invalidations=>2,
516 * :constant_cache_misses=>14,
517 * :class_serial=>546,
518 * :global_cvar_state=>27
519 * }
520 *
521 * The contents of the hash are implementation specific and may be changed in
522 * the future.
523 *
524 * This method is only expected to work on C Ruby.
525 */
526static VALUE
527vm_stat(int argc, VALUE *argv, VALUE self)
528{
529 static VALUE sym_constant_cache_invalidations, sym_constant_cache_misses, sym_class_serial, sym_global_cvar_state;
530 VALUE arg = Qnil;
531 VALUE hash = Qnil, key = Qnil;
532
533 if (rb_check_arity(argc, 0, 1) == 1) {
534 arg = argv[0];
535 if (SYMBOL_P(arg))
536 key = arg;
537 else if (RB_TYPE_P(arg, T_HASH))
538 hash = arg;
539 else
540 rb_raise(rb_eTypeError, "non-hash or symbol given");
541 }
542 else {
543 hash = rb_hash_new();
544 }
545
546#define S(s) sym_##s = ID2SYM(rb_intern_const(#s))
547 S(constant_cache_invalidations);
548 S(constant_cache_misses);
549 S(class_serial);
550 S(global_cvar_state);
551#undef S
552
553#define SET(name, attr) \
554 if (key == sym_##name) \
555 return SERIALT2NUM(attr); \
556 else if (hash != Qnil) \
557 rb_hash_aset(hash, sym_##name, SERIALT2NUM(attr));
558
559 SET(constant_cache_invalidations, ruby_vm_constant_cache_invalidations);
560 SET(constant_cache_misses, ruby_vm_constant_cache_misses);
561 SET(class_serial, ruby_vm_class_serial);
562 SET(global_cvar_state, ruby_vm_global_cvar_state);
563#undef SET
564
565 if (!NIL_P(key)) { /* matched key should return above */
566 rb_raise(rb_eArgError, "unknown key: %"PRIsVALUE, rb_sym2str(key));
567 }
568
569 return hash;
570}

768static VALUE
769vm_make_env_each(const rb_execution_context_t * const ec, rb_control_frame_t *const cfp)
770{
771 const VALUE * const ep = cfp->ep;
772 const rb_env_t *env;
773 const rb_iseq_t *env_iseq;
774 VALUE *env_body, *env_ep;
775 int local_size, env_size;
776
777 if (VM_ENV_ESCAPED_P(ep)) {
778 return VM_ENV_ENVVAL(ep);
779 }
780
781 if (!VM_ENV_LOCAL_P(ep)) {
782 const VALUE *prev_ep = VM_ENV_PREV_EP(ep);
783 if (!VM_ENV_ESCAPED_P(prev_ep)) {
784 rb_control_frame_t *prev_cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(cfp);
785
786 while (prev_cfp->ep != prev_ep) {
787 prev_cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(prev_cfp);
788 VM_ASSERT(prev_cfp->ep != NULL);
789 }
790
791 vm_make_env_each(ec, prev_cfp);
792 VM_FORCE_WRITE_SPECIAL_CONST(&ep[VM_ENV_DATA_INDEX_SPECVAL], VM_GUARDED_PREV_EP(prev_cfp->ep));
793 }
794 }
795 else {
796 VALUE block_handler = VM_ENV_BLOCK_HANDLER(ep);
797
798 if (block_handler != VM_BLOCK_HANDLER_NONE) {
799 VALUE blockprocval = vm_block_handler_escape(ec, block_handler);
800 VM_STACK_ENV_WRITE(ep, VM_ENV_DATA_INDEX_SPECVAL, blockprocval);
801 }
802 }
803
804 if (!VM_FRAME_RUBYFRAME_P(cfp)) {
805 local_size = VM_ENV_DATA_SIZE;
806 }
807 else {
808 local_size = ISEQ_BODY(cfp->iseq)->local_table_size + VM_ENV_DATA_SIZE;
809 }
810
811 /*
812 * # local variables on a stack frame (N == local_size)
813 * [lvar1, lvar2, ..., lvarN, SPECVAL]
814 * ^
815 * ep[0]
816 *
817 * # moved local variables
818 * [lvar1, lvar2, ..., lvarN, SPECVAL, Envval, BlockProcval (if needed)]
819 * ^ ^
820 * env->env[0] ep[0]
821 */
822
823 env_size = local_size +
824 1 /* envval */;
825 env_body = ALLOC_N(VALUE, env_size);
826 MEMCPY(env_body, ep - (local_size - 1 /* specval */), VALUE, local_size);
827
828#if 0
829 for (i = 0; i < local_size; i++) {
830 if (VM_FRAME_RUBYFRAME_P(cfp)) {
831 /* clear value stack for GC */
832 ep[-local_size + i] = 0;
833 }
834 }
835#endif
836
837 env_iseq = VM_FRAME_RUBYFRAME_P(cfp) ? cfp->iseq : NULL;
838 env_ep = &env_body[local_size - 1 /* specval */];
839
840 env = vm_env_new(env_ep, env_body, env_size, env_iseq);
841
842 cfp->ep = env_ep;
843 VM_ENV_FLAGS_SET(env_ep, VM_ENV_FLAG_ESCAPED | VM_ENV_FLAG_WB_REQUIRED);
844 VM_STACK_ENV_WRITE(ep, 0, (VALUE)env); /* GC mark */
845 return (VALUE)env;
846}

1034static enum rb_id_table_iterator_result
1035collect_outer_variable_names(ID id, VALUE val, void *ptr)
1036{
1037 struct collect_outer_variable_name_data *data = (struct collect_outer_variable_name_data *)ptr;
1038
1039 if (id == rb_intern("yield")) {
1040 data->yield = true;
1041 }
1042 else {

1045 val == Qtrue /* write */) {

1057static const rb_env_t *
1058env_copy(const VALUE *src_ep, VALUE read_only_variables)
1059{
1060 const rb_env_t *src_env = (rb_env_t *)VM_ENV_ENVVAL(src_ep);
1061 VM_ASSERT(src_env->ep == src_ep);
1062
1063 VALUE *env_body = ZALLOC_N(VALUE, src_env->env_size); // fill with Qfalse
1064 VALUE *ep = &env_body[src_env->env_size - 2];
1065 volatile VALUE prev_env = Qnil;
1066
1067 if (read_only_variables) {
1068 for (int i=RARRAY_LENINT(read_only_variables)-1; i>=0; i--) {
1069 ID id = NUM2ID(RARRAY_AREF(read_only_variables, i));
1070
1071 for (unsigned int j=0; j<ISEQ_BODY(src_env->iseq)->local_table_size; j++) {
1072 if (id == ISEQ_BODY(src_env->iseq)->local_table[j]) {
1073 VALUE v = src_env->env[j];
1074 if (!rb_ractor_shareable_p(v)) {
1075 VALUE name = rb_id2str(id);
1076 VALUE msg = rb_sprintf("can not make shareable Proc because it can refer"
1077 " unshareable object %+" PRIsVALUE " from ", v);
1078 if (name)
1079 rb_str_catf(msg, "variable `%" PRIsVALUE "'", name);
1080 else
1081 rb_str_cat_cstr(msg, "a hidden variable");
1082 rb_exc_raise(rb_exc_new_str(rb_eRactorIsolationError, msg));
1083 }
1084 env_body[j] = v;
1085 rb_ary_delete_at(read_only_variables, i);
1086 break;
1087 }
1088 }
1089 }
1090 }
1091
1092 ep[VM_ENV_DATA_INDEX_ME_CREF] = src_ep[VM_ENV_DATA_INDEX_ME_CREF];
1093 ep[VM_ENV_DATA_INDEX_FLAGS] = src_ep[VM_ENV_DATA_INDEX_FLAGS] | VM_ENV_FLAG_ISOLATED;
1094
1095 if (!VM_ENV_LOCAL_P(src_ep)) {
1096 const VALUE *prev_ep = VM_ENV_PREV_EP(src_env->ep);
1097 const rb_env_t *new_prev_env = env_copy(prev_ep, read_only_variables);
1098 prev_env = (VALUE)new_prev_env;
1099 ep[VM_ENV_DATA_INDEX_SPECVAL] = VM_GUARDED_PREV_EP(new_prev_env->ep);
1100 }
1101 else {
1102 ep[VM_ENV_DATA_INDEX_SPECVAL] = VM_BLOCK_HANDLER_NONE;
1103 }
1104
1105 const rb_env_t *copied_env = vm_env_new(ep, env_body, src_env->env_size, src_env->iseq);
1106 RB_GC_GUARD(prev_env);
1107 return copied_env;
1108}

1277 VALUE idtmp = 0;
1278 rb_ast_id_table_t *dyns = ALLOCV(idtmp, sizeof(rb_ast_id_table_t) + dyncount * sizeof(ID));
1279 dyns->size = dyncount;
1280 MEMCPY(dyns->ids, dynvars, ID, dyncount);
1281
1282 rb_node_init(&tmp_node, NODE_SCOPE, (VALUE)dyns, 0, 0);
1283 ast.root = &tmp_node;
1284 ast.compile_option = 0;
1285 ast.script_lines = INT2FIX(-1);
1286
1287 if (base_iseq) {
1288 iseq = rb_iseq_new(&ast, ISEQ_BODY(base_iseq)->location.label, path, realpath, base_iseq, ISEQ_TYPE_EVAL);
1289 }
1290 else {
1291 VALUE tempstr = rb_fstring_lit("<temp>");
1292 iseq = rb_iseq_new_top(&ast, tempstr, tempstr, tempstr, NULL);
1293 }
1294 tmp_node.nd_tbl = 0; /* reset table */
1295 ALLOCV_END(idtmp);
1296
1297 vm_set_eval_stack(ec, iseq, 0, base_block);
1298 vm_bind_update_env(bindval, bind, envval = vm_make_env_object(ec, ec->cfp));
1299 rb_vm_pop_frame(ec);
1300
1301 env = (const rb_env_t *)envval;
1302 return env->env;
1303}

1347 VALUE self, int argc, const VALUE *argv, int kw_splat, VALUE passed_block_handler,

1350static inline VALUE
1351invoke_iseq_block_from_c(rb_execution_context_t *ec, const struct rb_captured_block *captured,
1352 VALUE self, int argc, const VALUE *argv, int kw_splat, VALUE passed_block_handler,
1353 const rb_cref_t *cref, int is_lambda, const rb_callable_method_entry_t *me)
1354{
1355 const rb_iseq_t *iseq = rb_iseq_check(captured->code.iseq);
1356 int i, opt_pc;
1357 VALUE type = VM_FRAME_MAGIC_BLOCK | (is_lambda ? VM_FRAME_FLAG_LAMBDA : 0);
1358 rb_control_frame_t *cfp = ec->cfp;
1359 VALUE *sp = cfp->sp;
1360
1361 stack_check(ec);
1362
1363 CHECK_VM_STACK_OVERFLOW(cfp, argc);
1364 vm_check_canary(ec, sp);
1365 cfp->sp = sp + argc;
1366 for (i=0; i<argc; i++) {
1367 sp[i] = argv[i];
1368 }
1369
1370 opt_pc = vm_yield_setup_args(ec, iseq, argc, sp, kw_splat, passed_block_handler,
1371 (is_lambda ? arg_setup_method : arg_setup_block));
1372 cfp->sp = sp;
1373
1374 if (me == NULL) {
1375 return invoke_block(ec, iseq, self, captured, cref, type, opt_pc);
1376 }
1377 else {
1378 return invoke_bmethod(ec, iseq, self, captured, me, type, opt_pc);
1379 }

1410 }

2039// Whether cfp is the last frame in the unwinding process for a non-local return.
2040static void

2042 bool cfp_returning_with_value, int state, struct vm_throw_data *err)
2043{
2044 if (state == TAG_RAISE && RBASIC(err)->klass == rb_eSysStackError) {
2045 return;
2046 }
2047 else {
2048 const rb_iseq_t *iseq = cfp->iseq;

2050
2051 switch (VM_FRAME_TYPE(ec->cfp)) {
2052 case VM_FRAME_MAGIC_METHOD:
2053 RUBY_DTRACE_METHOD_RETURN_HOOK(ec, 0, 0);
2054 EXEC_EVENT_HOOK_AND_POP_FRAME(ec, RUBY_EVENT_RETURN, ec->cfp->self, 0, 0, 0, frame_return_value(err));
2055
2056 if (UNLIKELY(local_hooks && local_hooks->events & RUBY_EVENT_RETURN)) {

2074 rb_exec_event_hook_orig(ec, local_hooks, RUBY_EVENT_B_RETURN,
2075 ec->cfp->self, 0, 0, 0, bmethod_return_value, FALSE);
2076 }

2097 }
2098 else {
2099 EXEC_EVENT_HOOK_AND_POP_FRAME(ec, RUBY_EVENT_B_RETURN, ec->cfp->self, 0, 0, 0, frame_return_value(err));

2110 }

2498
2499 /* push block frame */
2500 cfp->sp[0] = (VALUE)err;
2501 vm_push_frame(ec, catch_iseq, VM_FRAME_MAGIC_RESCUE,
2502 cfp->self,
2503 VM_GUARDED_PREV_EP(cfp->ep),
2504 0, /* cref or me */
2505 ISEQ_BODY(catch_iseq)->iseq_encoded,
2506 cfp->sp + arg_size /* push value */,
2507 ISEQ_BODY(catch_iseq)->local_table_size - arg_size,
2508 ISEQ_BODY(catch_iseq)->stack_max);
2509
2510 state = 0;
2511 ec->tag->state = TAG_NONE;
2512 ec->errinfo = Qnil;
2513
2514 return Qundef;
2515 }
2516 else {
2517 hook_before_rewind(ec, ec->cfp, (cfp == escape_cfp), state, err);
2518
2519 if (VM_FRAME_FINISHED_P(ec->cfp)) {
2520 rb_vm_pop_frame(ec);
2521 ec->errinfo = (VALUE)err;
2522 ec->tag = ec->tag->prev;

3632 * Document-class: Thread
3633 *
3634 * Threads are the Ruby implementation for a concurrent programming model.
3635 *
3636 * Programs that require multiple threads of execution are a perfect
3637 * candidate for Ruby's Thread class.
3638 *
3639 * For example, we can create a new thread separate from the main thread's
3640 * execution using ::new.
3641 *
3642 * thr = Thread.new { puts "What's the big deal" }
3643 *
3644 * Then we are able to pause the execution of the main thread and allow
3645 * our new thread to finish, using #join:
3646 *
3647 * thr.join #=> "What's the big deal"
3648 *
3649 * If we don't call +thr.join+ before the main thread terminates, then all
3650 * other threads including +thr+ will be killed.
3651 *
3652 * Alternatively, you can use an array for handling multiple threads at
3653 * once, like in the following example:
3654 *
3655 * threads = []
3656 * threads << Thread.new { puts "What's the big deal" }
3657 * threads << Thread.new { 3.times { puts "Threads are fun!" } }
3658 *
3659 * After creating a few threads we wait for them all to finish
3660 * consecutively.
3661 *
3662 * threads.each { |thr| thr.join }
3663 *
3664 * To retrieve the last value of a thread, use #value
3665 *
3666 * thr = Thread.new { sleep 1; "Useful value" }
3667 * thr.value #=> "Useful value"
3668 *
3669 * === Thread initialization
3670 *
3671 * In order to create new threads, Ruby provides ::new, ::start, and
3672 * ::fork. A block must be provided with each of these methods, otherwise
3673 * a ThreadError will be raised.
3674 *
3675 * When subclassing the Thread class, the +initialize+ method of your
3676 * subclass will be ignored by ::start and ::fork. Otherwise, be sure to
3677 * call super in your +initialize+ method.
3678 *
3679 * === Thread termination
3680 *
3681 * For terminating threads, Ruby provides a variety of ways to do this.
3682 *
3683 * The class method ::kill, is meant to exit a given thread:
3684 *
3685 * thr = Thread.new { sleep }
3686 * Thread.kill(thr) # sends exit() to thr
3687 *
3688 * Alternatively, you can use the instance method #exit, or any of its
3689 * aliases #kill or #terminate.
3690 *
3691 * thr.exit
3692 *
3693 * === Thread status
3694 *
3695 * Ruby provides a few instance methods for querying the state of a given
3696 * thread. To get a string with the current thread's state use #status
3697 *
3698 * thr = Thread.new { sleep }
3699 * thr.status # => "sleep"
3700 * thr.exit
3701 * thr.status # => false
3702 *
3703 * You can also use #alive? to tell if the thread is running or sleeping,
3704 * and #stop? if the thread is dead or sleeping.
3705 *
3706 * === Thread variables and scope
3707 *
3708 * Since threads are created with blocks, the same rules apply to other
3709 * Ruby blocks for variable scope. Any local variables created within this
3710 * block are accessible to only this thread.
3711 *
3712 * ==== Fiber-local vs. Thread-local
3713 *
3714 * Each fiber has its own bucket for Thread#[] storage. When you set a
3715 * new fiber-local it is only accessible within this Fiber. To illustrate:
3716 *
3717 * Thread.new {
3718 * Thread.current[:foo] = "bar"
3719 * Fiber.new {
3720 * p Thread.current[:foo] # => nil
3721 * }.resume
3722 * }.join
3723 *
3724 * This example uses #[] for getting and #[]= for setting fiber-locals,
3725 * you can also use #keys to list the fiber-locals for a given
3726 * thread and #key? to check if a fiber-local exists.
3727 *
3728 * When it comes to thread-locals, they are accessible within the entire
3729 * scope of the thread. Given the following example:
3730 *
3731 * Thread.new{
3732 * Thread.current.thread_variable_set(:foo, 1)
3733 * p Thread.current.thread_variable_get(:foo) # => 1
3734 * Fiber.new{
3735 * Thread.current.thread_variable_set(:foo, 2)
3736 * p Thread.current.thread_variable_get(:foo) # => 2
3737 * }.resume
3738 * p Thread.current.thread_variable_get(:foo) # => 2
3739 * }.join
3740 *
3741 * You can see that the thread-local +:foo+ carried over into the fiber
3742 * and was changed to +2+ by the end of the thread.
3743 *
3744 * This example makes use of #thread_variable_set to create new
3745 * thread-locals, and #thread_variable_get to reference them.
3746 *
3747 * There is also #thread_variables to list all thread-locals, and
3748 * #thread_variable? to check if a given thread-local exists.
3749 *
3750 * === Exception handling
3751 *
3752 * When an unhandled exception is raised inside a thread, it will
3753 * terminate. By default, this exception will not propagate to other
3754 * threads. The exception is stored and when another thread calls #value
3755 * or #join, the exception will be re-raised in that thread.
3756 *
3757 * t = Thread.new{ raise 'something went wrong' }
3758 * t.value #=> RuntimeError: something went wrong
3759 *
3760 * An exception can be raised from outside the thread using the
3761 * Thread#raise instance method, which takes the same parameters as
3762 * Kernel#raise.
3763 *
3764 * Setting Thread.abort_on_exception = true, Thread#abort_on_exception =
3765 * true, or $DEBUG = true will cause a subsequent unhandled exception
3766 * raised in a thread to be automatically re-raised in the main thread.
3767 *
3768 * With the addition of the class method ::handle_interrupt, you can now
3769 * handle exceptions asynchronously with threads.
3770 *
3771 * === Scheduling
3772 *
3773 * Ruby provides a few ways to support scheduling threads in your program.
3774 *
3775 * The first way is by using the class method ::stop, to put the current
3776 * running thread to sleep and schedule the execution of another thread.
3777 *
3778 * Once a thread is asleep, you can use the instance method #wakeup to
3779 * mark your thread as eligible for scheduling.
3780 *
3781 * You can also try ::pass, which attempts to pass execution to another
3782 * thread but is dependent on the OS whether a running thread will switch
3783 * or not. The same goes for #priority, which lets you hint to the thread
3784 * scheduler which threads you want to take precedence when passing
3785 * execution. This method is also dependent on the OS and may be ignored
3786 * on some platforms.
3787 *
3788 */
