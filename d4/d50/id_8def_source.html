145def id2varname(token, prefix = nil)
146 if /#/ =~ token
147 token = "_#{token.gsub(/\W+/, '_')}"
148 else
149 token = token.sub(/\?/, 'P')
150 token = prefix + token if prefix
151 token.sub!(/\A[a-z]/) {$&.upcase}
152 token.sub!(/\A\$/, "_G_")
153 token.sub!(/\A@@/, "_C_")
154 token.sub!(/\A@/, "_I_")
155 token.gsub!(/\W+/, "")
156 end
157 token
158end

160predefined_ids = {}
161preserved_ids = []
162local_ids = []
163instance_ids = []
164global_ids = []
165const_ids = []
166class_ids = []
167attrset_ids = []
168token_op_ids = []
169names = {}
170predefined.split(/^/).each_with_index do |line, num|
171 next if /^#/ =~ line
172 line.sub!(/\s+#.*/, '')
173 name, token = line.split
174 next unless name
175 token = id2varname(token || name)
176 if name == '-'
177 preserved_ids << token
178 next
179 end
180 if prev = names[name]
181 KeywordError.raise("#{name} is already registered at line #{prev+firstline}", firstline+num)
182 end
183 if prev = predefined_ids[token]
184 KeywordError.raise("#{token} is already used for #{prev} at line #{names[prev]+firstline}", firstline+num)
185 end
186 names[name] = num
187 case name
188 when /\A[A-Z]\w*\z/; const_ids
189 when /\A(?!\d)\w+\z/; local_ids
190 when /\A\$(?:\d+|(?!\d)\w+|\W)\z/; global_ids
191 when /\A@@(?!\d)\w+\z/; class_ids
192 when /\A@(?!\d)\w+\z/; instance_ids
193 when /\A((?!\d)\w+)=\z/; attrset_ids
194 else preserved_ids
195 end << token
196 predefined_ids[token] = name
197end
198token_ops.split(/^/).each do |line|
199 next if /^#/ =~ line
200 line.sub!(/\s+#.*/, '')
201 id, op, token = line.split
202 next unless id and op
203 token ||= (id unless /\A\W\z/ =~ op)
204 token_op_ids << [id, op, token]
205end
206{
207 "LOCAL" => local_ids,
208 "INSTANCE" => instance_ids,
209 "GLOBAL" => global_ids,
210 "CONST" => const_ids,
211 "CLASS" => class_ids,
