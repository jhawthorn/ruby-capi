178int
179ruby_env_debug_option(const char *str, int len, void *arg)
180{
181 int ov;
182 size_t retlen;
183 unsigned long n;
184#define SET_WHEN(name, var, val) do { \
185 if (len == sizeof(name) - 1 && \
186 strncmp(str, (name), len) == 0) { \
187 (var) = (val); \
188 return 1; \
189 } \
190 } while (0)
191#define NAME_MATCH_VALUE(name) \
192 ((size_t)len >= sizeof(name)-1 && \
193 strncmp(str, (name), sizeof(name)-1) == 0 && \
194 ((len == sizeof(name)-1 && !(len = 0)) || \
195 (str[sizeof(name)-1] == '=' && \
196 (str += sizeof(name), len -= sizeof(name), 1))))
197#define SET_UINT(val) do { \
198 n = ruby_scan_digits(str, len, 10, &retlen, &ov); \
199 if (!ov && retlen) { \
200 val = (unsigned int)n; \
201 } \
202 str += retlen; \
203 len -= retlen; \
204 } while (0)
205#define SET_UINT_LIST(name, vals, num) do { \
206 int i; \
207 for (i = 0; i < (num); ++i) { \
208 SET_UINT((vals)[i]); \
209 if (!len || *str != ':') break; \
210 ++str; \
211 --len; \
212 } \
213 if (len > 0) { \
214 fprintf(stderr, "ignored "name" option: `%.*s'\n", len, str); \
215 } \
216 } while (0)
217#define SET_WHEN_UINT(name, vals, num, req) \
218 if (NAME_MATCH_VALUE(name)) SET_UINT_LIST(name, vals, num);
219
220 SET_WHEN("gc_stress", *ruby_initial_gc_stress_ptr, Qtrue);
221 SET_WHEN("core", ruby_enable_coredump, 1);
222 SET_WHEN("ci", ruby_on_ci, 1);
223 if (NAME_MATCH_VALUE("rgengc")) {
224 if (!len) ruby_rgengc_debug = 1;
225 else SET_UINT_LIST("rgengc", &ruby_rgengc_debug, 1);
226 return 1;
227 }
228#if defined _WIN32
229# if RUBY_MSVCRT_VERSION >= 80
230 SET_WHEN("rtc_error", ruby_w32_rtc_error, 1);
231# endif
232#endif
233#if defined _WIN32 || defined __CYGWIN__
234 if (NAME_MATCH_VALUE("codepage")) {
235 if (!len) fprintf(stderr, "missing codepage argument");
236 else SET_UINT_LIST("codepage", ruby_w32_codepage, numberof(ruby_w32_codepage));
237 return 1;
238 }
239#endif
240 return 0;
241}

307static enum debug_log_filter_type
308filter_type(const char *str, int *skiplen)
309{
310 if (strncmp(str, "file:", 5) == 0) {
311 *skiplen = 5;
312 return dlf_file;
313 }
314 else if(strncmp(str, "func:", 5) == 0) {
315 *skiplen = 5;
316 return dlf_func;
317 }
318 else {
319 *skiplen = 0;
320 return dlf_all;

324static void
325setup_debug_log_filter(void)
326{
327 const char *filter_config = getenv("RUBY_DEBUG_LOG_FILTER");
328
329 if (filter_config && strlen(filter_config) > 0) {
330 unsigned int i;
331 for (i=0; i<MAX_DEBUG_LOG_FILTER_NUM && filter_config; i++) {
332 size_t len;
333 const char *str = filter_config;
334 const char *p;
335
336 if ((p = strchr(str, ',')) == NULL) {
337 len = strlen(str);
338 filter_config = NULL;
339 }
340 else {
341 len = p - str - 1; // 1 is ','
342 filter_config = p + 1;
343 }
344
345 // positive/negative
346 if (*str == '-') {
347 debug_log.filters[i].negative = true;
348 str++;
349 } else if (*str == '+') {
350 // negative is false on default.
351 str++;
352 }
353
354 // type
355 int skiplen;
356 debug_log.filters[i].type = filter_type(str, &skiplen);
357 len -= skiplen;
358
359 if (len >= MAX_DEBUG_LOG_FILTER_LEN) {
360 fprintf(stderr, "too long: %s (max:%d)\n", str, MAX_DEBUG_LOG_FILTER_LEN - 1);
361 exit(1);
362 }
363
364 // body
365 strncpy(debug_log.filters[i].str, str + skiplen, len);
366 debug_log.filters[i].str[len] = 0;
367 }
368 debug_log.filters_num = i;
369
370 for (i=0; i<debug_log.filters_num; i++) {
371 fprintf(stderr, "RUBY_DEBUG_LOG_FILTER[%d]=%s (%s%s)\n", i,
372 debug_log.filters[i].str,
373 debug_log.filters[i].negative ? "-" : "",
374 dlf_type_names[debug_log.filters[i].type]);
375 }
376 }
377}

379static void
380setup_debug_log(void)
381{
382 // check RUBY_DEBUG_LOG
383 const char *log_config = getenv("RUBY_DEBUG_LOG");
384 if (log_config && strlen(log_config) > 0) {
385 if (strcmp(log_config, "mem") == 0) {
386 debug_log.mem = (char *)malloc(MAX_DEBUG_LOG * MAX_DEBUG_LOG_MESSAGE_LEN);
387 if (debug_log.mem == NULL) {
388 fprintf(stderr, "setup_debug_log failed (can't allocate memory)\n");
389 exit(1);
390 }
391 ruby_debug_log_mode |= ruby_debug_log_memory;
392 }
393 else if (strcmp(log_config, "stderr") == 0) {
394 ruby_debug_log_mode |= ruby_debug_log_stderr;
395 }
396 else {
397 ruby_debug_log_mode |= ruby_debug_log_file;
398 if ((debug_log.output = fopen(log_config, "w")) == NULL) {
399 fprintf(stderr, "can not open %s for RUBY_DEBUG_LOG\n", log_config);
400 exit(1);
401 }
402 setvbuf(debug_log.output, NULL, _IONBF, 0);
403 }
404
405 fprintf(stderr, "RUBY_DEBUG_LOG=%s %s%s%s\n", log_config,
406 (ruby_debug_log_mode & ruby_debug_log_memory) ? "[mem]" : "",
407 (ruby_debug_log_mode & ruby_debug_log_stderr) ? "[stderr]" : "",
408 (ruby_debug_log_mode & ruby_debug_log_file) ? "[file]" : "");
409 rb_nativethread_lock_initialize(&debug_log.lock);
410
411 setup_debug_log_filter();
412 }
413}

415static bool
416check_filter(const char *str, const struct debug_log_filter *filter, bool *state)
417{
418 if (filter->negative) {
419 if (strstr(str, filter->str) == NULL) {
420 *state = true;
421 return false;
422 }
423 else {
424 *state = false;
425 return true;
426 }
427 }
428 else {
429 if (strstr(str, filter->str) != NULL) {
430 *state = true;
431 return true;
432 }
433 else {
434 *state = false;
435 return false;
436 }
437 }
438}

440//
441// RUBY_DEBUG_LOG_FILTER=-foo,-bar,baz,boo
442// returns true if
443// (func_name or file_name) doesn't contain foo
444// and
445// (func_name or file_name) doesn't contain bar
446// and
447// (func_name or file_name) contains baz or boo
448//
449// RUBY_DEBUG_LOG_FILTER=foo,bar,-baz,-boo
450// retunrs true if
451// (func_name or file_name) contains foo or bar
452// or
453// (func_name or file_name) doesn't contain baz and
454// (func_name or file_name) doesn't contain boo and
455//
456// You can specify "file:" (ex file:foo) or "func:" (ex func:foo)
457// prefixes to specify the filter for.
458//
459bool
460ruby_debug_log_filter(const char *func_name, const char *file_name)
461{
462 if (debug_log.filters_num > 0) {
463 bool state = false;
464
465 for (unsigned int i = 0; i<debug_log.filters_num; i++) {
466 const struct debug_log_filter *filter = &debug_log.filters[i];
467
468 switch (filter->type) {
469 case dlf_all:
470 if (check_filter(func_name, filter, &state)) return state;
471 if (check_filter(file_name, filter, &state)) return state;
472 break;
473 case dlf_func:
474 if (check_filter(func_name, filter, &state)) return state;
475 break;
476 case dlf_file:
477 if (check_filter(file_name, filter, &state)) return state;
478 break;
479 }
480 }
481 return state;
482 }
483 else {
484 return true;
485 }
486}

499void
500ruby_debug_log(const char *file, int line, const char *func_name, const char *fmt, ...)
501{
502 char buff[MAX_DEBUG_LOG_MESSAGE_LEN] = {0};
503 int len = 0;
504 int r = 0;
505
506 // message title
507 if (func_name && len < MAX_DEBUG_LOG_MESSAGE_LEN) {
508 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN, "%s\t", func_name);
509 if (r < 0) rb_bug("ruby_debug_log returns %d\n", r);
510 len += r;
511 }
512
513 // message
514 if (fmt && len < MAX_DEBUG_LOG_MESSAGE_LEN) {
515 va_list args;
516 va_start(args, fmt);
517 r = vsnprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN - len, fmt, args);
518 va_end(args);
519 if (r < 0) rb_bug("ruby_debug_log vsnprintf() returns %d", r);
520 len += r;
521 }
522
523 // optional information
524
525 // C location
526 if (file && len < MAX_DEBUG_LOG_MESSAGE_LEN) {
527 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN, "\t%s:%d", pretty_filename(file), line);
528 if (r < 0) rb_bug("ruby_debug_log returns %d\n", r);
529 len += r;
530 }
531
532 if (rb_current_execution_context(false)) {
533 // Ruby location
534 int ruby_line;
535 const char *ruby_file = rb_source_location_cstr(&ruby_line);
536 if (len < MAX_DEBUG_LOG_MESSAGE_LEN) {
537 if (ruby_file) {
538 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN - len, "\t%s:%d", pretty_filename(ruby_file), ruby_line);
539 }
540 else {
541 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN - len, "\t");
542 }
543 if (r < 0) rb_bug("ruby_debug_log returns %d\n", r);
544 len += r;
545 }
546
547 // ractor information
548 if (ruby_single_main_ractor == NULL) {
549 rb_ractor_t *cr = GET_RACTOR();
550 if (r && len < MAX_DEBUG_LOG_MESSAGE_LEN) {
551 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN - len, "\tr:#%u/%u",
552 (unsigned int)rb_ractor_id(cr), GET_VM()->ractor.cnt);
553 if (r < 0) rb_bug("ruby_debug_log returns %d\n", r);
554 len += r;
555 }
556 }
557
558 // thread information
559 const rb_thread_t *th = GET_THREAD();
560 if (r && len < MAX_DEBUG_LOG_MESSAGE_LEN) {
561 r = snprintf(buff + len, MAX_DEBUG_LOG_MESSAGE_LEN - len, "\tth:%u", rb_th_serial(th));
562 if (r < 0) rb_bug("ruby_debug_log returns %d\n", r);
563 len += r;
564 }
565 }
566
567 rb_nativethread_lock_lock(&debug_log.lock);
568 {
569 unsigned int cnt = debug_log.cnt++;
570
571 if (ruby_debug_log_mode & ruby_debug_log_memory) {
572 unsigned int index = cnt % MAX_DEBUG_LOG;
573 char *dst = RUBY_DEBUG_LOG_MEM_ENTRY(index);
574 strncpy(dst, buff, MAX_DEBUG_LOG_MESSAGE_LEN);
575 }
576 if (ruby_debug_log_mode & ruby_debug_log_stderr) {
577 fprintf(stderr, "%4u: %s\n", cnt, buff);
578 }
579 if (ruby_debug_log_mode & ruby_debug_log_file) {
580 fprintf(debug_log.output, "%u\t%s\n", cnt, buff);
581 }
582 }
583 rb_nativethread_lock_unlock(&debug_log.lock);
584}

586// for debugger
587static void
588debug_log_dump(FILE *out, unsigned int n)
589{
590 if (ruby_debug_log_mode & ruby_debug_log_memory) {
591 unsigned int size = debug_log.cnt > MAX_DEBUG_LOG ? MAX_DEBUG_LOG : debug_log.cnt;
592 unsigned int current_index = debug_log.cnt % MAX_DEBUG_LOG;
593 if (n == 0) n = size;
594 if (n > size) n = size;
595
596 for (unsigned int i=0; i<n; i++) {
597 int index = current_index - size + i;
598 if (index < 0) index += MAX_DEBUG_LOG;
599 VM_ASSERT(index <= MAX_DEBUG_LOG);
600 const char *mesg = RUBY_DEBUG_LOG_MEM_ENTRY(index);;
601 fprintf(out, "%4u: %s\n", debug_log.cnt - size + i, mesg);
602 }
603 }
604 else {
605 fprintf(stderr, "RUBY_DEBUG_LOG=mem is not specified.");
606 }
607}
