274static inline int
275iseq_catch_table_bytes(int n)
276{
277 enum {
278 catch_table_entry_size = sizeof(struct iseq_catch_table_entry),
279 catch_table_entries_max = (INT_MAX - offsetof(struct iseq_catch_table, entries)) / catch_table_entry_size
280 };
281 if (n > catch_table_entries_max) rb_fatal("too large iseq_catch_table - %d", n);
282 return (int)(offsetof(struct iseq_catch_table, entries) +
283 n * catch_table_entry_size);
284}
