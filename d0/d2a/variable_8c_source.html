113static VALUE
114classname(VALUE klass, int *permanent)
115{
116 st_table *ivtbl;
117 st_data_t n;
118
119 *permanent = 0;
120 if (!RCLASS_EXT(klass)) return Qnil;
121 if (!(ivtbl = RCLASS_IV_TBL(klass))) return Qnil;
122 if (st_lookup(ivtbl, (st_data_t)classpath, &n)) {
123 *permanent = 1;
124 return (VALUE)n;
125 }
126 if (st_lookup(ivtbl, (st_data_t)tmp_classpath, &n)) return (VALUE)n;
127 return Qnil;
128}

144static VALUE
145make_temporary_path(VALUE obj, VALUE klass)
146{
147 VALUE path;
148 switch (klass) {
149 case Qnil:
150 path = rb_sprintf("#<Class:%p>", (void*)obj);
151 break;
152 case Qfalse:
153 path = rb_sprintf("#<Module:%p>", (void*)obj);
154 break;
155 default:
156 path = rb_sprintf("#<%"PRIsVALUE":%p>", klass, (void*)obj);
157 break;
158 }
159 OBJ_FREEZE(path);
160 return path;

165static VALUE
166rb_tmp_class_path(VALUE klass, int *permanent, fallback_func fallback)
167{
168 VALUE path = classname(klass, permanent);
169
170 if (!NIL_P(path)) {
171 return path;
172 }
173 else {
174 if (RB_TYPE_P(klass, T_MODULE)) {
175 if (rb_obj_class(klass) == rb_cModule) {
176 path = Qfalse;
177 }
178 else {
179 int perm;
180 path = rb_tmp_class_path(RBASIC(klass)->klass, &perm, fallback);
181 }
182 }
183 *permanent = 0;
184 return fallback(klass, path);
185 }

259VALUE
260rb_path_to_class(VALUE pathname)
261{
262 rb_encoding *enc = rb_enc_get(pathname);
263 const char *pbeg, *pend, *p, *path = RSTRING_PTR(pathname);
264 ID id;
265 VALUE c = rb_cObject;
266
267 if (!rb_enc_asciicompat(enc)) {
268 rb_raise(rb_eArgError, "invalid class path encoding (non ASCII)");
269 }
270 pbeg = p = path;
271 pend = path + RSTRING_LEN(pathname);
272 if (path == pend || path[0] == '#') {
273 rb_raise(rb_eArgError, "can't retrieve anonymous class %"PRIsVALUE,
274 QUOTE(pathname));
275 }
276 while (p < pend) {
277 while (p < pend && *p != ':') p++;
278 id = rb_check_id_cstr(pbeg, p-pbeg, enc);
279 if (p < pend && p[0] == ':') {
280 if ((size_t)(pend - p) < 2 || p[1] != ':') goto undefined_class;
281 p += 2;
282 pbeg = p;
283 }
284 if (!id) {
285 goto undefined_class;
286 }
287 c = rb_const_search(c, id, TRUE, FALSE, FALSE);
288 if (c == Qundef) goto undefined_class;
289 if (!rb_namespace_p(c)) {
290 rb_raise(rb_eTypeError, "%"PRIsVALUE" does not refer to class/module",
291 pathname);
292 }
293 }
294 RB_GC_GUARD(pathname);
295
296 return c;
297
298 undefined_class:
299 rb_raise(rb_eArgError, "undefined class/module % "PRIsVALUE,
300 rb_str_subseq(pathname, 0, p-path));
301 UNREACHABLE_RETURN(Qundef);
302}

355static struct rb_global_entry*
356rb_find_global_entry(ID id)
357{
358 struct rb_global_entry *entry;
359 VALUE data;
360
361 if (!rb_id_table_lookup(rb_global_tbl, id, &data)) {
362 entry = NULL;
363 }
364 else {
365 entry = (struct rb_global_entry *)data;
366 RUBY_ASSERT(entry != NULL);
367 }
368
369 if (UNLIKELY(!rb_ractor_main_p()) && (!entry || !entry->ractor_local)) {
370 rb_raise(rb_eRactorIsolationError, "can not access global variables %s from non-main Ractors", rb_id2name(id));
371 }
372
373 return entry;
374}

388static struct rb_global_entry*
389rb_global_entry(ID id)
390{
391 struct rb_global_entry *entry = rb_find_global_entry(id);
392 if (!entry) {

394 entry = ALLOC(struct rb_global_entry);

397 entry->var = var;
398 entry->ractor_local = false;

405
406 var->block_trace = 0;
407 var->trace = 0;
408 rb_id_table_insert(rb_global_tbl, id, (VALUE)entry);
409 }
410 return entry;

500{

540static ID
541global_id(const char *name)
542{
543 ID id;
544
545 if (name[0] == '$') id = rb_intern(name);
546 else {
547 size_t len = strlen(name);
548 VALUE vbuf = 0;
549 char *buf = ALLOCV_N(char, vbuf, len+1);
550 buf[0] = '$';
551 memcpy(buf+1, name, len);
552 id = rb_intern2(buf, len+1);
553 ALLOCV_END(vbuf);
554 }
555 return id;
556}

627VALUE
628rb_f_trace_var(int argc, const VALUE *argv)
629{
630 VALUE var, cmd;
631 struct rb_global_entry *entry;
632 struct trace_var *trace;
633
634 if (rb_scan_args(argc, argv, "11", &var, &cmd) == 1) {
635 cmd = rb_block_proc();
636 }
637 if (NIL_P(cmd)) {
638 return rb_f_untrace_var(argc, argv);
639 }
640 entry = rb_global_entry(rb_to_id(var));
641 trace = ALLOC(struct trace_var);
642 trace->next = entry->var->trace;
643 trace->func = rb_trace_eval;
644 trace->data = cmd;
645 trace->removed = 0;
646 entry->var->trace = trace;
647
648 return Qnil;
649}

651static void
652remove_trace(struct rb_global_variable *var)
653{
654 struct trace_var *trace = var->trace;
655 struct trace_var t;
656 struct trace_var *next;
657
658 t.next = trace;
659 trace = &t;
660 while (trace->next) {
661 next = trace->next;
662 if (next->removed) {
663 trace->next = next->next;
664 xfree(next);
665 }
666 else {
667 trace = next;
668 }
669 }
670 var->trace = t.next;
671}

673VALUE
674rb_f_untrace_var(int argc, const VALUE *argv)
675{
676 VALUE var, cmd;
677 ID id;
678 struct rb_global_entry *entry;
679 struct trace_var *trace;
680
681 rb_scan_args(argc, argv, "11", &var, &cmd);
682 id = rb_check_id(&var);
683 if (!id) {
684 rb_name_error_str(var, "undefined global variable %"PRIsVALUE"", QUOTE(var));
685 }
686 if ((entry = rb_find_global_entry(id)) == NULL) {
687 rb_name_error(id, "undefined global variable %"PRIsVALUE"", QUOTE_ID(id));
688 }
689
690 trace = entry->var->trace;
691 if (NIL_P(cmd)) {
692 VALUE ary = rb_ary_new();
693
694 while (trace) {
695 struct trace_var *next = trace->next;
696 rb_ary_push(ary, (VALUE)trace->data);
697 trace->removed = 1;
698 trace = next;
699 }
700
701 if (!entry->var->block_trace) remove_trace(entry->var);
702 return ary;
703 }
704 else {
705 while (trace) {
706 if (trace->data == cmd) {
707 trace->removed = 1;
708 if (!entry->var->block_trace) remove_trace(entry->var);
709 return rb_ary_new3(1, cmd);
710 }
711 trace = trace->next;
712 }
713 }
714 return Qnil;
715}

886 xfree(trace);
887 trace = next;

890 }

1254rb_ivar_lookup(VALUE obj, ID id, VALUE undef)
1255{
1256 if (SPECIAL_CONST_P(obj)) return undef;
1257 switch (BUILTIN_TYPE(obj)) {
1258 case T_OBJECT:
1259 {

1266 index < len &&

1270 else {

1273 }

1277 st_data_t val;

1282 UNLIKELY(!rb_ractor_main_p()) &&

1285 "can not get unshareable values from instance variables of classes/modules from non-main Ractors");
1286 }
1287 return val;
1288 }
1289 else {
1290 break;
1291 }
1292 }
1293 default:
1294 if (FL_TEST(obj, FL_EXIVAR))
1295 return generic_ivar_get(obj, id, undef);

1297 }

1933/*
1934 * call-seq:
1935 * obj.remove_instance_variable(symbol) -> obj
1936 * obj.remove_instance_variable(string) -> obj
1937 *
1938 * Removes the named instance variable from <i>obj</i>, returning that
1939 * variable's value.
1940 * String arguments are converted to symbols.
1941 *
1942 * class Dummy
1943 * attr_reader :var
1944 * def initialize
1945 * @var = 99
1946 * end
1947 * def remove
1948 * remove_instance_variable(:@var)
1949 * end
1950 * end
1951 * d = Dummy.new
1952 * d.var #=> 99
1953 * d.remove #=> 99
1954 * d.var #=> nil
1955 */
1956
1957VALUE
1958rb_obj_remove_instance_variable(VALUE obj, VALUE name)
1959{
1960 VALUE val = Qnil;
1961 const ID id = id_for_var(obj, name, an, instance);
1962 st_data_t n, v;
1963 struct st_table *iv_index_tbl;
1964 uint32_t index;
1965
1966 rb_check_frozen(obj);
1967 if (!id) {
1968 goto not_defined;
1969 }
1970
1971 switch (BUILTIN_TYPE(obj)) {
1972 case T_OBJECT:
1973 iv_index_tbl = ROBJECT_IV_INDEX_TBL(obj);
1974 if (iv_index_tbl_lookup(iv_index_tbl, id, &index) &&
1975 index < ROBJECT_NUMIV(obj) &&
1976 (val = ROBJECT_IVPTR(obj)[index]) != Qundef) {
1977 ROBJECT_IVPTR(obj)[index] = Qundef;
1978 return val;
1979 }
1980 break;
1981 case T_CLASS:
1982 case T_MODULE:
1983 IVAR_ACCESSOR_SHOULD_BE_MAIN_RACTOR(id);
1984 n = id;
1985 if (RCLASS_IV_TBL(obj) && lock_st_delete(RCLASS_IV_TBL(obj), &n, &v)) {
1986 return (VALUE)v;
1987 }
1988 break;
1989 default:
1990 if (FL_TEST(obj, FL_EXIVAR)) {
1991 if (generic_ivar_remove(obj, id, &val)) {
1992 return val;
1993 }
1994 }
1995 break;
1996 }

2000 obj, name);

2002}

2028 *
2029 * Invoked when a reference is made to an undefined constant in
2030 * <i>mod</i>. It is passed a symbol for the undefined constant, and

2032 * following code is an example of the same:
2033 *
2034 * def Foo.const_missing(name)
2035 * name # return the constant name as Symbol
2036 * end
2037 *
2038 * Foo::UNDEFINED_CONST #=> :UNDEFINED_CONST: symbol returned
2039 *

2043 * <code>fred.rb</code>). If found, it returns the loaded class. It
2044 * therefore implements an autoload feature similar to Kernel#autoload and
2045 * Module#autoload.
2046 *
2047 * def Object.const_missing(name)
2048 * @looked_for ||= {}
2049 * str_name = name.to_s
2050 * raise "Class not found: #{name}" if @looked_for[str_name]
2051 * @looked_for[str_name] = 1
2052 * file = str_name.downcase
2053 * require file
2054 * klass = const_get(name)
2055 * return klass if klass
2056 * raise "Class not found: #{name}"
2057 * end

2684VALUE
2685rb_autoload_load(VALUE module, ID name)
2686{
2687 rb_const_entry_t *ce = rb_const_lookup(module, name);
2688
2689 // We bail out as early as possible without any synchronisation:
2690 if (!ce || ce->value != Qundef) {
2691 return Qfalse;
2692 }
2693
2694 // At this point, we assume there might be autoloading, so fail if it's ractor:
2695 if (UNLIKELY(!rb_ractor_main_p())) {
2696 rb_raise(rb_eRactorUnsafeError, "require by autoload on non-main Ractor is not supported (%s)", rb_id2name(name));
2697 }
2698
2699 // This state is stored on thes stack and is used during the autoload process.
2700 struct autoload_load_arguments arguments = {.module = module, .name = name, .mutex = Qnil};
2701
2702 // Figure out whether we can autoload the named constant:
2703 VALUE autoload_const_value = rb_mutex_synchronize(autoload_mutex, autoload_load_needed, (VALUE)&arguments);
2704
2705 // This confirms whether autoloading is required or not:
2706 if (autoload_const_value == Qfalse) return autoload_const_value;
2707
2708 arguments.flag = ce->flag & (CONST_DEPRECATED | CONST_VISIBILITY_MASK);
2709
2710 // Only one thread will enter here at a time:
2711 VALUE result = rb_mutex_synchronize(arguments.mutex, autoload_try_load, (VALUE)&arguments);
2712
2713 // If you don't guard this value, it's possible for the autoload constant to
2714 // be freed by another thread which loads multiple constants, one of which
2715 // resolves to the constant this thread is trying to load, so proteect this
2716 // so that it is not freed until we are done with it in `autoload_try_load`:
2717 RB_GC_GUARD(autoload_const_value);
2718
2719 return result;
2720}

2774static VALUE
2775rb_const_search_from(VALUE klass, ID id, int exclude, int recurse, int visibility)
2776{
2777 VALUE value, current;
2778 bool first_iteration = true;
2779
2780 for (current = klass;
2781 RTEST(current);
2782 current = RCLASS_SUPER(current), first_iteration = false) {
2783 VALUE tmp;
2784 VALUE am = 0;
2785 rb_const_entry_t *ce;
2786
2787 if (!first_iteration && RCLASS_ORIGIN(current) != current) {
2788 // This item in the super chain has an origin iclass
2789 // that comes later in the chain. Skip this item so
2790 // prepended modules take precedence.
2791 continue;
2792 }
2793
2794 // Do lookup in original class or module in case we are at an origin
2795 // iclass in the chain.
2796 tmp = current;
2797 if (BUILTIN_TYPE(tmp) == T_ICLASS) tmp = RBASIC(tmp)->klass;
2798
2799 // Do the lookup. Loop in case of autoload.
2800 while ((ce = rb_const_lookup(tmp, id))) {
2801 if (visibility && RB_CONST_PRIVATE_P(ce)) {
2802 GET_EC()->private_const_reference = tmp;
2803 return Qundef;
2804 }
2805 rb_const_warn_if_deprecated(ce, tmp, id);
2806 value = ce->value;
2807 if (value == Qundef) {
2808 struct autoload_const *ac;
2809 if (am == tmp) break;
2810 am = tmp;
2811 ac = autoloading_const_entry(tmp, id);
2812 if (ac) return ac->value;
2813 rb_autoload_load(tmp, id);
2814 continue;
2815 }
2816 if (exclude && tmp == rb_cObject) {
2817 goto not_found;
2818 }
2819 return value;
2820 }
2821 if (!recurse) break;
2822 }
2823
2824 not_found:
2825 GET_EC()->private_const_reference = 0;
2826 return Qundef;
2827}
