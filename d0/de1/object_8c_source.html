217MJIT_FUNC_EXPORTED void
218rb_obj_copy_ivar(VALUE dest, VALUE obj)
219{
220 VALUE *dst_buf = 0;
221 VALUE *src_buf = 0;
222 uint32_t len = ROBJECT_EMBED_LEN_MAX;
223
224 if (RBASIC(obj)->flags & ROBJECT_EMBED) {
225 src_buf = ROBJECT(obj)->as.ary;
226
227 // embedded -> embedded
228 if (RBASIC(dest)->flags & ROBJECT_EMBED) {
229 dst_buf = ROBJECT(dest)->as.ary;
230 }
231 // embedded -> extended
232 else {
233 dst_buf = ROBJECT(dest)->as.heap.ivptr;
234 }
235 }
236 // extended -> extended
237 else {
238 RUBY_ASSERT(!(RBASIC(dest)->flags & ROBJECT_EMBED));
239 uint32_t src_len = ROBJECT(obj)->as.heap.numiv;
240 uint32_t dst_len = ROBJECT(dest)->as.heap.numiv;
241
242 len = src_len < dst_len ? src_len : dst_len;
243 dst_buf = ROBJECT(dest)->as.heap.ivptr;
244 src_buf = ROBJECT(obj)->as.heap.ivptr;
245 }
246
247 MEMCPY(dst_buf, src_buf, VALUE, len);
248}

346static VALUE
347mutable_obj_clone(VALUE obj, VALUE kwfreeze)
348{
349 VALUE clone, singleton;
350 VALUE argv[2];
351
352 clone = rb_obj_alloc(rb_obj_class(obj));
353
354 singleton = rb_singleton_class_clone_and_attach(obj, clone);
355 RBASIC_SET_CLASS(clone, singleton);
356 if (FL_TEST(singleton, FL_SINGLETON)) {
357 rb_singleton_class_attached(singleton, clone);
358 }
359
360 init_copy(clone, obj);
361
362 switch (kwfreeze) {
363 case Qnil:
364 rb_funcall(clone, id_init_clone, 1, obj);
365 RBASIC(clone)->flags |= RBASIC(obj)->flags & FL_FREEZE;
366 break;
367 case Qtrue:
368 {
369 static VALUE freeze_true_hash;
370 if (!freeze_true_hash) {
371 freeze_true_hash = rb_hash_new();
372 rb_gc_register_mark_object(freeze_true_hash);
373 rb_hash_aset(freeze_true_hash, ID2SYM(idFreeze), Qtrue);
374 rb_obj_freeze(freeze_true_hash);
375 }
376
377 argv[0] = obj;
378 argv[1] = freeze_true_hash;
379 rb_funcallv_kw(clone, id_init_clone, 2, argv, RB_PASS_KEYWORDS);
380 RBASIC(clone)->flags |= FL_FREEZE;
381 break;
382 }
383 case Qfalse:
384 {
385 static VALUE freeze_false_hash;
386 if (!freeze_false_hash) {
387 freeze_false_hash = rb_hash_new();
388 rb_gc_register_mark_object(freeze_false_hash);
389 rb_hash_aset(freeze_false_hash, ID2SYM(idFreeze), Qfalse);
390 rb_obj_freeze(freeze_false_hash);
391 }
392
393 argv[0] = obj;
394 argv[1] = freeze_false_hash;
395 rb_funcallv_kw(clone, id_init_clone, 2, argv, RB_PASS_KEYWORDS);
396 break;
397 }
398 default:
399 rb_bug("invalid kwfreeze passed to mutable_obj_clone");
400 }
401
402 return clone;
403}

412/*
413 * call-seq:
414 * obj.dup -> an_object
415 *
416 * Produces a shallow copy of <i>obj</i>---the instance variables of
417 * <i>obj</i> are copied, but not the objects they reference.
418 *
419 * This method may have class-specific behavior. If so, that
420 * behavior will be documented under the #+initialize_copy+ method of
421 * the class.
422 *
423 * === on dup vs clone
424 *
425 * In general, #clone and #dup may have different semantics in
426 * descendant classes. While #clone is used to duplicate an object,
427 * including its internal state, #dup typically uses the class of the
428 * descendant object to create the new instance.
429 *
430 * When using #dup, any modules that the object has been extended with will not
431 * be copied.
432 *
433 * class Klass
434 * attr_accessor :str
435 * end
436 *
437 * module Foo
438 * def foo; 'foo'; end
439 * end
440 *
441 * s1 = Klass.new #=> #<Klass:0x401b3a38>
442 * s1.extend(Foo) #=> #<Klass:0x401b3a38>
443 * s1.foo #=> "foo"
444 *
445 * s2 = s1.clone #=> #<Klass:0x401be280>
446 * s2.foo #=> "foo"
447 *
448 * s3 = s1.dup #=> #<Klass:0x401c1084>
449 * s3.foo #=> NoMethodError: undefined method `foo' for #<Klass:0x401c1084>
450 */
451VALUE
452rb_obj_dup(VALUE obj)
453{
454 VALUE dup;
455
456 if (special_object_p(obj)) {
457 return obj;
458 }
459 dup = rb_obj_alloc(rb_obj_class(obj));
460 init_copy(dup, obj);
461 rb_funcall(dup, id_init_dup, 1, obj);
462
463 return dup;
464}

581static int
582inspect_i(st_data_t k, st_data_t v, st_data_t a)
583{
584 ID id = (ID)k;
585 VALUE value = (VALUE)v;
586 VALUE str = (VALUE)a;
587
588 /* need not to show internal data */
589 if (CLASS_OF(value) == 0) return ST_CONTINUE;
590 if (!rb_is_instance_id(id)) return ST_CONTINUE;
591 if (RSTRING_PTR(str)[0] == '-') { /* first element */
592 RSTRING_PTR(str)[0] = '#';
593 rb_str_cat2(str, " ");
594 }
595 else {
596 rb_str_cat2(str, ", ");
597 }
598 rb_str_catf(str, "%"PRIsVALUE"=%+"PRIsVALUE,
599 rb_id2str(id), value);
600
601 return ST_CONTINUE;
602}

622 *
623 * Returns a string containing a human-readable representation of <i>obj</i>.
624 * The default #inspect shows the object's class name, an encoding of

628 * <i>obj</i>. When overriding this method, it should return a string
629 * whose encoding is compatible with the default external encoding.
630 *
631 * [ 1, 2, 3..4, 'five' ].inspect #=> "[1, 2, 3..4, \"five\"]"
632 * Time.new.inspect #=> "2008-03-08 19:43:39 +0900"
633 *
634 * class Foo
635 * end
636 * Foo.new.inspect #=> "#<Foo:0x0300c868>"
637 *

639 * def initialize

641 * end
642 * end
643 * Bar.new.inspect #=> "#<Bar:0x0300c868 @bar=1>"

648{

650 VALUE str;

654 return rb_exec_recursive(inspect_obj, obj, str);

702// Returns whether c is a proper (c != cl) subclass of cl
703// Both c and cl must be T_CLASS
704static VALUE
705class_search_class_ancestor(VALUE cl, VALUE c)
706{
707 RUBY_ASSERT(RB_TYPE_P(c, T_CLASS));
708 RUBY_ASSERT(RB_TYPE_P(cl, T_CLASS));
709
710 size_t c_depth = RCLASS_SUPERCLASS_DEPTH(c);
711 size_t cl_depth = RCLASS_SUPERCLASS_DEPTH(cl);
712 VALUE *classes = RCLASS_SUPERCLASSES(cl);
713
714 // If c's inheritance chain is longer, it cannot be an ancestor
715 // We are checking for a proper subclass so don't check if they are equal
716 if (cl_depth <= c_depth)
717 return Qfalse;
718
719 // Otherwise check that c is in cl's inheritance chain
720 return RBOOL(classes[c_depth] == c);
721}

723/*
724 * call-seq:
725 * obj.is_a?(class) -> true or false
726 * obj.kind_of?(class) -> true or false
727 *
728 * Returns <code>true</code> if <i>class</i> is the class of
729 * <i>obj</i>, or if <i>class</i> is one of the superclasses of
730 * <i>obj</i> or modules included in <i>obj</i>.
731 *
732 * module M; end
733 * class A
734 * include M
735 * end
736 * class B < A; end
737 * class C < B; end
738 *
739 * b = B.new
740 * b.is_a? A #=> true
741 * b.is_a? B #=> true
742 * b.is_a? C #=> false
743 * b.is_a? M #=> true
744 *
745 * b.kind_of? A #=> true
746 * b.kind_of? B #=> true
747 * b.kind_of? C #=> false
748 * b.kind_of? M #=> true
749 */
750
751VALUE
752rb_obj_is_kind_of(VALUE obj, VALUE c)
753{
754 VALUE cl = CLASS_OF(obj);
755
756 RUBY_ASSERT(RB_TYPE_P(cl, T_CLASS));
757
758 // Fastest path: If the object's class is an exact match we know `c` is a
759 // class without checking type and can return immediately.
760 if (cl == c) return Qtrue;
761
762 // Note: YJIT needs this function to never allocate and never raise when
763 // `c` is a class or a module.
764
765 if (LIKELY(RB_TYPE_P(c, T_CLASS))) {
766 // Fast path: Both are T_CLASS
767 return class_search_class_ancestor(cl, c);
768 }
769 else if (RB_TYPE_P(c, T_ICLASS)) {
770 // First check if we inherit the includer
771 // If we do we can return true immediately
772 VALUE includer = RCLASS_INCLUDER(c);
773 if (cl == includer) return Qtrue;
774
775 // Usually includer is a T_CLASS here, except when including into an
776 // already included Module.
777 // If it is a class, attempt the fast class-to-class check and return
778 // true if there is a match.
779 if (RB_TYPE_P(includer, T_CLASS) && class_search_class_ancestor(cl, includer))
780 return Qtrue;
781
782 // We don't include the ICLASS directly, so must check if we inherit
783 // the module via another include
784 return RBOOL(class_search_ancestor(cl, RCLASS_ORIGIN(c)));
785 }
786 else if (RB_TYPE_P(c, T_MODULE)) {
787 // Slow path: check each ancestor in the linked list and its method table
788 return RBOOL(class_search_ancestor(cl, RCLASS_ORIGIN(c)));
789 }
790 else {
791 rb_raise(rb_eTypeError, "class or module required");
792 UNREACHABLE_RETURN(Qfalse);
793 }
794}

797static VALUE
798class_search_ancestor(VALUE cl, VALUE c)
799{
800 while (cl) {
801 if (cl == c || RCLASS_M_TBL(cl) == RCLASS_M_TBL(c))
802 return cl;
803 cl = RCLASS_SUPER(cl);
804 }
805 return 0;
806}

809VALUE
810rb_class_search_ancestor(VALUE cl, VALUE c)
811{

1563/*
1564 * call-seq:
1565 * mod <= other -> true, false, or nil
1566 *
1567 * Returns true if <i>mod</i> is a subclass of <i>other</i> or
1568 * is the same as <i>other</i>. Returns
1569 * <code>nil</code> if there's no relationship between the two.
1570 * (Think of the relationship in terms of the class definition:
1571 * "class A < B" implies "A < B".)
1572 */
1573
1574VALUE
1575rb_class_inherited_p(VALUE mod, VALUE arg)
1576{
1577 if (mod == arg) return Qtrue;
1578
1579 if (RB_TYPE_P(arg, T_CLASS) && RB_TYPE_P(mod, T_CLASS)) {
1580 // comparison between classes
1581 size_t mod_depth = RCLASS_SUPERCLASS_DEPTH(mod);
1582 size_t arg_depth = RCLASS_SUPERCLASS_DEPTH(arg);
1583 if (arg_depth < mod_depth) {
1584 // check if mod < arg
1585 return RCLASS_SUPERCLASSES(mod)[arg_depth] == arg ?
1586 Qtrue :
1587 Qnil;
1588 } else if (arg_depth > mod_depth) {
1589 // check if mod > arg
1590 return RCLASS_SUPERCLASSES(arg)[mod_depth] == mod ?
1591 Qfalse :
1592 Qnil;
1593 } else {
1594 // Depths match, and we know they aren't equal: no relation
1595 return Qnil;
1596 }
1597 } else {
1598 if (!CLASS_OR_MODULE_P(arg) && !RB_TYPE_P(arg, T_ICLASS)) {
1599 rb_raise(rb_eTypeError, "compared with non class/module");
1600 }
1601 if (class_search_ancestor(mod, RCLASS_ORIGIN(arg))) {
1602 return Qtrue;
1603 }
1604 /* not mod < arg; check if mod > arg */
1605 if (class_search_ancestor(arg, mod)) {
1606 return Qfalse;
1607 }
1608 return Qnil;
1609 }
1610}

2145 * the ancestors (and +Object+ if <i>mod</i> is a +Module+).
2146 *
2147 * The value of the constant is returned if a definition is found,
2148 * otherwise a +NameError+ is raised.
2149 *
2150 * Math.const_get(:PI) #=> 3.14159265358979
2151 *
2152 * This method will recursively look up constant names if a namespaced
2153 * class name is provided. For example:
2154 *
2155 * module Foo; class Bar; end end
2156 * Object.const_get 'Foo::Bar'
2157 *
2158 * The +inherit+ flag is respected on each lookup. For example:
2159 *
2160 * module Foo
2161 * class Bar
2162 * VAL = 10

2164 *
2165 * class Baz < Bar; end
2166 * end
2167 *
2168 * Object.const_get 'Foo::Baz::VAL' # => 10
2169 * Object.const_get 'Foo::Baz::VAL', false # => NameError
2170 *
2171 * If the argument is not a valid constant name a +NameError+ will be
2172 * raised with a warning "wrong constant name".
2173 *
2174 * Object.const_get 'foobar' #=> NameError: wrong constant name foobar
2175 *

3099 * Integer(object, base = 0, exception: true) -> integer or nil

3102 *

3104 * using +to_int+ first and +to_i+ second;
3105 * see below for exceptions.
3106 *
3107 * With integer argument +object+ given, returns +object+:
3108 *
3109 * Integer(1) # => 1
3110 * Integer(-1) # => -1
3111 *
3112 * With floating-point argument +object+ given,
3113 * returns +object+ truncated to an intger:
3114 *
3115 * Integer(1.9) # => 1 # Rounds toward zero.
3116 * Integer(-1.9) # => -1 # Rounds toward zero.
3117 *
3118 * With string argument +object+ and zero +base+ given,
3119 * returns +object+ converted to an integer in base 10:
3120 *
3121 * Integer('100') # => 100
3122 * Integer('-100') # => -100
3123 *
3124 * With +base+ zero, string +object+ may contain leading characters
3125 * to specify the actual base:
3126 *
3127 * Integer('0100') # => 64 # Leading '0' specifies base 8.
3128 * Integer('0b100') # => 4 # Leading '0b', specifies base 2.
3129 * Integer('0x100') # => 256 # Leading '0x' specifies base 16.
3130 *
3131 * With a non-zero +base+ (in range 2..36) given
3132 * (in which case +object+ must be a string),
3133 * returns +object+ converted to an integer in the given base:
3134 *
3135 * Integer('100', 2) # => 4
3136 * Integer('100', 8) # => 64
3137 * Integer('-100', 16) # => -256

3142 * Integer(' 100 ') # => 100
3143 * Integer('-1_0_0', 16) # => -256

3146 *
3147 * Integer(Rational(9, 10)) # => 0 # Rounds toward zero.
3148 * Integer(Complex(2, 0)) # => 2 # Imaginary part must be zero.

3852/* Document-class: BasicObject
3853 *
3854 * BasicObject is the parent class of all classes in Ruby. It's an explicit
3855 * blank class.
3856 *
3857 * BasicObject can be used for creating object hierarchies independent of
3858 * Ruby's object hierarchy, proxy objects like the Delegator class, or other
3859 * uses where namespace pollution from Ruby's methods and classes must be
3860 * avoided.
3861 *
3862 * To avoid polluting BasicObject for other users an appropriately named
3863 * subclass of BasicObject should be created instead of directly modifying
3864 * BasicObject:
3865 *
3866 * class MyObjectSystem < BasicObject
3867 * end
3868 *
3869 * BasicObject does not include Kernel (for methods like +puts+) and
3870 * BasicObject is outside of the namespace of the standard library so common
3871 * classes will not be found without using a full class path.
3872 *
3873 * A variety of strategies can be used to provide useful portions of the
3874 * standard library to subclasses of BasicObject. A subclass could
3875 * <code>include Kernel</code> to obtain +puts+, +exit+, etc. A custom
3876 * Kernel-like module could be created and included or delegation can be used
3877 * via #method_missing:
3878 *
3879 * class MyObjectSystem < BasicObject
3880 * DELEGATE = [:puts, :p]
3881 *
3882 * def method_missing(name, *args, &block)
3883 * return super unless DELEGATE.include? name
3884 * ::Kernel.send(name, *args, &block)
3885 * end
3886 *
3887 * def respond_to_missing?(name, include_private = false)
3888 * DELEGATE.include?(name) or super
3889 * end
3890 * end
3891 *
3892 * Access to classes and modules from the Ruby standard library can be
3893 * obtained in a BasicObject subclass by referencing the desired constant
3894 * from the root like <code>::File</code> or <code>::Enumerator</code>.
3895 * Like #method_missing, #const_missing can be used to delegate constant
3896 * lookup to +Object+:
3897 *
3898 * class MyObjectSystem < BasicObject
3899 * def self.const_missing(name)
3900 * ::Object.const_get(name)
3901 * end
3902 * end

3904 * === What's Here
3905 *
3906 * These are the methods defined for \BasicObject:
3907 *
3908 * - ::new: Returns a new \BasicObject instance.
3909 * - #!: Returns the boolean negation of +self+: +true+ or +false+.
3910 * - #!=: Returns whether +self+ and the given object are _not_ equal.
3911 * - #==: Returns whether +self+ and the given object are equivalent.
3912 * - #__id__: Returns the integer object identifier for +self+.
3913 * - #__send__: Calls the method identified by the given symbol.
3914 * - #equal?: Returns whether +self+ and the given object are the same object.
3915 * - #instance_eval: Evaluates the given string or block in the context of +self+.
