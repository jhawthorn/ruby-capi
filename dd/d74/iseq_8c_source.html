106 IC ic;
107};

109// This iterator is used to walk through the instructions and clean any
110// references to ICs that are contained within this ISEQ out of the VM's
111// constant cache table. It passes around a struct that holds the current IC
112// we're looking for, which can be NULL (if we haven't hit an opt_getinlinecache
113// instruction yet) or set to an IC (if we've hit an opt_getinlinecache and
114// haven't yet hit the associated opt_setinlinecache).
115static bool
116iseq_clear_ic_references_i(VALUE *code, VALUE insn, size_t index, void *data)
117{
118 struct iseq_clear_ic_references_data *ic_data = (struct iseq_clear_ic_references_data *) data;
119
120 switch (insn) {
121 case BIN(opt_getinlinecache): {
122 RUBY_ASSERT_ALWAYS(ic_data->ic == NULL);
123
124 ic_data->ic = (IC) code[index + 2];
125 return true;
126 }
127 case BIN(getconstant): {
128 if (ic_data->ic != NULL) {
129 ID id = (ID) code[index + 1];
130 rb_vm_t *vm = GET_VM();
131 VALUE lookup_result;
132
133 if (rb_id_table_lookup(vm->constant_cache, id, &lookup_result)) {
134 st_table *ics = (st_table *)lookup_result;
135 st_data_t ic = (st_data_t)ic_data->ic;
136 st_delete(ics, &ic, NULL);
137
138 if (ics->num_entries == 0) {
139 rb_id_table_delete(vm->constant_cache, id);
140 st_free_table(ics);
141 }
142 }
143 }
144
145 return true;
146 }
147 case BIN(opt_setinlinecache): {
148 RUBY_ASSERT_ALWAYS(ic_data->ic != NULL);
149
150 ic_data->ic = NULL;
151 return true;
152 }
153 default:
154 return true;
155 }
156}

158// When an ISEQ is being freed, all of its associated ICs are going to go away
159// as well. Because of this, we need to walk through the ISEQ, find any
160// opt_getinlinecache calls, and clear out the VM's constant cache of associated
161// ICs.
162static void
163iseq_clear_ic_references(const rb_iseq_t *iseq)
164{
165 struct iseq_clear_ic_references_data data = { .ic = NULL };
166 rb_iseq_each(iseq, 0, iseq_clear_ic_references_i, (void *) &data);
167}

170rb_iseq_free(const rb_iseq_t *iseq)
171{
172 RUBY_FREE_ENTER("iseq");
173
174 if (iseq && ISEQ_BODY(iseq)) {
175 iseq_clear_ic_references(iseq);
176 struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
177 mjit_free_iseq(iseq); /* Notify MJIT */
178#if YJIT_BUILD
179 rb_yjit_iseq_free(body->yjit_payload);
180#endif
181 ruby_xfree((void *)body->iseq_encoded);
182 ruby_xfree((void *)body->insns_info.body);
183 if (body->insns_info.positions) ruby_xfree((void *)body->insns_info.positions);
184#if VM_INSN_INFO_TABLE_IMPL == 2
185 if (body->insns_info.succ_index_table) ruby_xfree(body->insns_info.succ_index_table);
186#endif
187 if (LIKELY(body->local_table != rb_iseq_shared_exc_local_tbl))
188 ruby_xfree((void *)body->local_table);
189 ruby_xfree((void *)body->is_entries);
190
191 if (body->call_data) {
192 ruby_xfree(body->call_data);
193 }
194 ruby_xfree((void *)body->catch_table);
195 ruby_xfree((void *)body->param.opt_table);
196
197 if (body->param.keyword != NULL) {
198 ruby_xfree((void *)body->param.keyword->default_values);
199 ruby_xfree((void *)body->param.keyword);
200 }
201 compile_data_free(ISEQ_COMPILE_DATA(iseq));
202 if (body->outer_variables) rb_id_table_free(body->outer_variables);
203 ruby_xfree(body);
204 }
205
206 if (iseq && ISEQ_EXECUTABLE_P(iseq) && iseq->aux.exec.local_hooks) {
207 rb_hook_list_free(iseq->aux.exec.local_hooks);
208 }
209
210 RUBY_FREE_LEAVE("iseq");
211}

227// The translator for OPT_DIRECT_THREADED_CODE and OPT_CALL_THREADED_CODE does
228// some normalization to always return the non-trace version of instructions. To
229// mirror that behavior in token-threaded environments, we normalize in this
230// translator by also returning non-trace opcodes.
231static VALUE
232rb_vm_insn_normalizing_translator(const void *addr)
233{
234 VALUE opcode = (VALUE)addr;
235 VALUE trace_opcode_threshold = (VM_INSTRUCTION_SIZE / 2);
236
237 if (opcode >= trace_opcode_threshold) {
238 return opcode - trace_opcode_threshold;
239 }
240 return opcode;
241}

246static int
247iseq_extract_values(VALUE *code, size_t pos, iseq_value_itr_t * func, void *data, rb_vm_insns_translator_t * translator)
248{
249 VALUE insn = translator((void *)code[pos]);
250 int len = insn_len(insn);
251 int op_no;
252 const char *types = insn_op_types(insn);
253
254 for (op_no = 0; types[op_no]; op_no++) {
255 char type = types[op_no];
256 switch (type) {
257 case TS_CDHASH:
258 case TS_ISEQ:
259 case TS_VALUE:
260 {
261 VALUE op = code[pos + op_no + 1];
262 if (!SPECIAL_CONST_P(op)) {
263 VALUE newop = func(data, op);
264 if (newop != op) {
265 code[pos + op_no + 1] = newop;
266 }
267 }
268 }
269 break;
270 case TS_IC:
271 {
272 IC ic = (IC)code[pos + op_no + 1];
273 if (ic->entry) {
274 VALUE nv = func(data, (VALUE)ic->entry);
275 if ((VALUE)ic->entry != nv) {
276 ic->entry = (void *)nv;
277 }
278 }
279 }
280 break;
281 case TS_IVC:
282 case TS_ICVARC:
283 {
284 IVC ivc = (IVC)code[pos + op_no + 1];
285 if (ivc->entry) {
286 if (RB_TYPE_P(ivc->entry->class_value, T_NONE)) {
287 rb_bug("!! %u", ivc->entry->index);
288 }
289 VALUE nv = func(data, ivc->entry->class_value);
290 if (ivc->entry->class_value != nv) {
291 ivc->entry->class_value = nv;
292 }
293 }
294 }
295 break;
296 case TS_ISE:
297 {
298 union iseq_inline_storage_entry *const is = (union iseq_inline_storage_entry *)code[pos + op_no + 1];
299 if (is->once.value) {
300 VALUE nv = func(data, is->once.value);
301 if (is->once.value != nv) {
302 is->once.value = nv;
303 }
304 }
305 }
306 break;
307 default:
308 break;
309 }
310 }
311
312 return len;
313}

315static void
316rb_iseq_each_value(const rb_iseq_t *iseq, iseq_value_itr_t * func, void *data)
317{
318 unsigned int size;
319 VALUE *code;
320 size_t n;
321 rb_vm_insns_translator_t *const translator =
322#if OPT_DIRECT_THREADED_CODE || OPT_CALL_THREADED_CODE
323 (FL_TEST((VALUE)iseq, ISEQ_TRANSLATED)) ? rb_vm_insn_addr2insn2 :
324#endif
325 rb_vm_insn_null_translator;
326 const struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
327
328 size = body->iseq_size;
329 code = body->iseq_encoded;
330
331 for (n = 0; n < size;) {
332 n += iseq_extract_values(code, n, func, data, translator);
333 }
334}

336// Similar to rb_iseq_each_value, except that this walks through each
337// instruction instead of the associated VALUEs. The provided iterator should
338// return a boolean that indicates whether or not to continue iterating.
339void
340rb_iseq_each(const rb_iseq_t *iseq, size_t start_index, rb_iseq_each_i iterator, void *data)
341{
342 unsigned int size;
343 VALUE *code;
344 size_t index;
345
346 rb_vm_insns_translator_t *const translator =
347#if OPT_DIRECT_THREADED_CODE || OPT_CALL_THREADED_CODE
348 (FL_TEST((VALUE)iseq, ISEQ_TRANSLATED)) ? rb_vm_insn_addr2insn2 :
349#endif
350 rb_vm_insn_normalizing_translator; // Always pass non-trace opcodes.
351
352 const struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
353
354 size = body->iseq_size;
355 code = body->iseq_encoded;
356
357 for (index = start_index; index < size;) {
358 void *addr = (void *) code[index];
359 VALUE insn = translator(addr);
360
361 if (!iterator(code, insn, index, data)) {
362 break;
363 }
364
365 index += insn_len(insn);
366 }
367}

375void
376rb_iseq_update_references(rb_iseq_t *iseq)
377{
378 if (ISEQ_BODY(iseq)) {
379 struct rb_iseq_constant_body *body = ISEQ_BODY(iseq);
380
381 body->variable.coverage = rb_gc_location(body->variable.coverage);
382 body->variable.pc2branchindex = rb_gc_location(body->variable.pc2branchindex);
383 body->variable.script_lines = rb_gc_location(body->variable.script_lines);
384 body->location.label = rb_gc_location(body->location.label);
385 body->location.base_label = rb_gc_location(body->location.base_label);
386 body->location.pathobj = rb_gc_location(body->location.pathobj);
387 if (body->local_iseq) {
388 body->local_iseq = (struct rb_iseq_struct *)rb_gc_location((VALUE)body->local_iseq);
389 }
390 if (body->parent_iseq) {
391 body->parent_iseq = (struct rb_iseq_struct *)rb_gc_location((VALUE)body->parent_iseq);
392 }
393 if (body->mandatory_only_iseq) {
394 body->mandatory_only_iseq = (struct rb_iseq_struct *)rb_gc_location((VALUE)body->mandatory_only_iseq);
395 }
396 if (body->call_data) {
397 for (unsigned int i=0; i<body->ci_size; i++) {
398 struct rb_call_data *cds = body->call_data;
399 if (!SPECIAL_CONST_P((VALUE)cds[i].ci)) {
400 cds[i].ci = (struct rb_callinfo *)rb_gc_location((VALUE)cds[i].ci);
401 }
402 cds[i].cc = (struct rb_callcache *)rb_gc_location((VALUE)cds[i].cc);
403 }
404 }
405 if (FL_TEST((VALUE)iseq, ISEQ_MARKABLE_ISEQ)) {
406 rb_iseq_each_value(iseq, update_each_insn_value, NULL);
407 VALUE *original_iseq = ISEQ_ORIGINAL_ISEQ(iseq);
408 if (original_iseq) {
409 size_t n = 0;
410 const unsigned int size = body->iseq_size;
411 while (n < size) {
412 n += iseq_extract_values(original_iseq, n, update_each_insn_value, NULL, rb_vm_insn_null_translator);
413 }
414 }
415 }
416
417 if (body->param.flags.has_kw && ISEQ_COMPILE_DATA(iseq) == NULL) {
418 int i, j;
419
420 i = body->param.keyword->required_num;
421
422 for (j = 0; i < body->param.keyword->num; i++, j++) {
423 VALUE obj = body->param.keyword->default_values[j];
424 if (obj != Qundef) {
425 body->param.keyword->default_values[j] = rb_gc_location(obj);
426 }
427 }
428 }
429
430 if (body->catch_table) {
431 struct iseq_catch_table *table = body->catch_table;
432 unsigned int i;
433 for (i = 0; i < table->size; i++) {
434 struct iseq_catch_table_entry *entry;
435 entry = UNALIGNED_MEMBER_PTR(table, entries[i]);
436 if (entry->iseq) {
437 entry->iseq = (rb_iseq_t *)rb_gc_location((VALUE)entry->iseq);
438 }
439 }
440 }
441#if USE_MJIT
442 mjit_update_references(iseq);
443#endif
444#if YJIT_BUILD
445 rb_yjit_iseq_update_references(body->yjit_payload);
446#endif
447 }
448}

457void
458rb_iseq_mark(const rb_iseq_t *iseq)
459{
460 RUBY_MARK_ENTER("iseq");
461
462 RUBY_MARK_UNLESS_NULL(iseq->wrapper);
463
464 if (ISEQ_BODY(iseq)) {
465 const struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
466
467 if (FL_TEST((VALUE)iseq, ISEQ_MARKABLE_ISEQ)) {
468 rb_iseq_each_value(iseq, each_insn_value, NULL);
469 }
470
471 rb_gc_mark_movable(body->variable.coverage);
472 rb_gc_mark_movable(body->variable.pc2branchindex);
473 rb_gc_mark_movable(body->variable.script_lines);
474 rb_gc_mark_movable(body->location.label);
475 rb_gc_mark_movable(body->location.base_label);
476 rb_gc_mark_movable(body->location.pathobj);
477 RUBY_MARK_MOVABLE_UNLESS_NULL((VALUE)body->mandatory_only_iseq);
478 RUBY_MARK_MOVABLE_UNLESS_NULL((VALUE)body->parent_iseq);
479
480 if (body->call_data) {
481 struct rb_call_data *cds = (struct rb_call_data *)body->call_data;
482 for (unsigned int i=0; i<body->ci_size; i++) {
483 const struct rb_callinfo *ci = cds[i].ci;
484 const struct rb_callcache *cc = cds[i].cc;
485
486 if (vm_ci_markable(ci)) {
487 rb_gc_mark_movable((VALUE)ci);
488 }
489
490 if (cc) {
491 VM_ASSERT((cc->flags & VM_CALLCACHE_ON_STACK) == 0);
492
493 if (vm_cc_markable(cc)) {
494 if (!vm_cc_invalidated_p(cc)) {
495 rb_gc_mark_movable((VALUE)cc);
496 }
497 else {
498 cds[i].cc = rb_vm_empty_cc();
499 }
500 }
501 }
502 }
503 }
504
505 if (body->param.flags.has_kw && ISEQ_COMPILE_DATA(iseq) == NULL) {
506 const struct rb_iseq_param_keyword *const keyword = body->param.keyword;
507 int i, j;
508
509 i = keyword->required_num;
510
511 for (j = 0; i < keyword->num; i++, j++) {
512 VALUE obj = keyword->default_values[j];
513 if (!SPECIAL_CONST_P(obj)) {
514 rb_gc_mark_movable(obj);
515 }
516 }
517 }
518
519 if (body->catch_table) {
520 const struct iseq_catch_table *table = body->catch_table;
521 unsigned int i;
522 for (i = 0; i < table->size; i++) {
523 const struct iseq_catch_table_entry *entry;
524 entry = UNALIGNED_MEMBER_PTR(table, entries[i]);
525 if (entry->iseq) {
526 rb_gc_mark_movable((VALUE)entry->iseq);
527 }
528 }
529 }
530
531#if USE_MJIT
532 mjit_mark_cc_entries(body);
533#endif
534#if YJIT_BUILD
535 rb_yjit_iseq_mark(body->yjit_payload);
536#endif
537 }
538
539 if (FL_TEST_RAW((VALUE)iseq, ISEQ_NOT_LOADED_YET)) {
540 rb_gc_mark(iseq->aux.loader.obj);
541 }
542 else if (FL_TEST_RAW((VALUE)iseq, ISEQ_USE_COMPILE_DATA)) {
543 const struct iseq_compile_data *const compile_data = ISEQ_COMPILE_DATA(iseq);
544
545 rb_iseq_mark_insn_storage(compile_data->insn.storage_head);
546
547 RUBY_MARK_UNLESS_NULL(compile_data->err_info);
548 if (RTEST(compile_data->catch_table_ary)) {
549 rb_gc_mark(compile_data->catch_table_ary);
550 }
551 VM_ASSERT(compile_data != NULL);
552 }
553 else {
554 /* executable */
555 VM_ASSERT(ISEQ_EXECUTABLE_P(iseq));
556 if (iseq->aux.exec.local_hooks) {
557 rb_hook_list_mark(iseq->aux.exec.local_hooks);
558 }
559 }
560
561 RUBY_MARK_LEAVE("iseq");
562}

577size_t
578rb_iseq_memsize(const rb_iseq_t *iseq)
579{
580 size_t size = 0; /* struct already counted as RVALUE size */
581 const struct rb_iseq_constant_body *body = ISEQ_BODY(iseq);
582 const struct iseq_compile_data *compile_data;
583
584 /* TODO: should we count original_iseq? */
585
586 if (ISEQ_EXECUTABLE_P(iseq) && body) {
587 size += sizeof(struct rb_iseq_constant_body);
588 size += body->iseq_size * sizeof(VALUE);
589 size += body->insns_info.size * (sizeof(struct iseq_insn_info_entry) + sizeof(unsigned int));
590 size += body->local_table_size * sizeof(ID);
591 if (body->catch_table) {
592 size += iseq_catch_table_bytes(body->catch_table->size);
593 }
594 size += (body->param.opt_num + 1) * sizeof(VALUE);
595 size += param_keyword_size(body->param.keyword);
596
597 /* body->is_entries */
598 size += body->is_size * sizeof(union iseq_inline_storage_entry);
599
600 /* body->call_data */
601 size += body->ci_size * sizeof(struct rb_call_data);
602 // TODO: should we count imemo_callinfo?
603 }
604
605 compile_data = ISEQ_COMPILE_DATA(iseq);
606 if (compile_data) {
607 struct iseq_compile_data_storage *cur;
608
609 size += sizeof(struct iseq_compile_data);
610
611 cur = compile_data->node.storage_head;
612 while (cur) {
613 size += cur->size + offsetof(struct iseq_compile_data_storage, buff);
614 cur = cur->next;
615 }
616 }
617
618 return size;
619}

664iseq_location_setup(rb_iseq_t *iseq, VALUE name, VALUE path, VALUE realpath, VALUE first_lineno, const rb_code_location_t *code_location, const int node_id)

669 RB_OBJ_WRITE(iseq, &loc->label, name);

727static VALUE
728prepare_iseq_build(rb_iseq_t *iseq,
729 VALUE name, VALUE path, VALUE realpath, VALUE first_lineno, const rb_code_location_t *code_location, const int node_id,
730 const rb_iseq_t *parent, int isolated_depth, enum iseq_type type,
731 VALUE script_lines, const rb_compile_option_t *option)
732{
733 VALUE coverage = Qfalse;
734 VALUE err_info = Qnil;
735 struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
736
737 if (parent && (type == ISEQ_TYPE_MAIN || type == ISEQ_TYPE_TOP))
738 err_info = Qfalse;
739
740 body->type = type;
741 set_relation(iseq, parent);
742
743 name = rb_fstring(name);
744 iseq_location_setup(iseq, name, path, realpath, first_lineno, code_location, node_id);
745 if (iseq != body->local_iseq) {
746 RB_OBJ_WRITE(iseq, &body->location.base_label, ISEQ_BODY(body->local_iseq)->location.label);
747 }
748 ISEQ_COVERAGE_SET(iseq, Qnil);
749 ISEQ_ORIGINAL_ISEQ_CLEAR(iseq);
750 body->variable.flip_count = 0;
751
752 if (NIL_P(script_lines)) {
753 RB_OBJ_WRITE(iseq, &body->variable.script_lines, Qnil);
754 }
755 else {
756 RB_OBJ_WRITE(iseq, &body->variable.script_lines, rb_ractor_make_shareable(script_lines));
757 }
758
759 ISEQ_COMPILE_DATA_ALLOC(iseq);
760 RB_OBJ_WRITE(iseq, &ISEQ_COMPILE_DATA(iseq)->err_info, err_info);
761 RB_OBJ_WRITE(iseq, &ISEQ_COMPILE_DATA(iseq)->catch_table_ary, Qnil);
762
763 ISEQ_COMPILE_DATA(iseq)->node.storage_head = ISEQ_COMPILE_DATA(iseq)->node.storage_current = new_arena();
764 ISEQ_COMPILE_DATA(iseq)->insn.storage_head = ISEQ_COMPILE_DATA(iseq)->insn.storage_current = new_arena();
765 ISEQ_COMPILE_DATA(iseq)->isolated_depth = isolated_depth;
766 ISEQ_COMPILE_DATA(iseq)->option = option;
767 ISEQ_COMPILE_DATA(iseq)->ivar_cache_table = NULL;
768 ISEQ_COMPILE_DATA(iseq)->builtin_function_table = GET_VM()->builtin_function_table;
769
770
771 if (option->coverage_enabled) {
772 VALUE coverages = rb_get_coverages();
773 if (RTEST(coverages)) {
774 coverage = rb_hash_lookup(coverages, rb_iseq_path(iseq));
775 if (NIL_P(coverage)) coverage = Qfalse;
776 }
777 }
778 ISEQ_COVERAGE_SET(iseq, coverage);
779 if (coverage && ISEQ_BRANCH_COVERAGE(iseq))
780 ISEQ_PC2BRANCHINDEX_SET(iseq, rb_ary_tmp_new(0));
781
782 return Qtrue;
783}

1119static VALUE
1120iseq_load(VALUE data, const rb_iseq_t *parent, VALUE opt)
1121{
1122 rb_iseq_t *iseq = iseq_alloc();
1123
1124 VALUE magic, version1, version2, format_type, misc;
1125 VALUE name, path, realpath, first_lineno, code_location, node_id;
1126 VALUE type, body, locals, params, exception;
1127
1128 st_data_t iseq_type;
1129 rb_compile_option_t option;
1130 int i = 0;
1131 rb_code_location_t tmp_loc = { {0, 0}, {-1, -1} };
1132
1133 /* [magic, major_version, minor_version, format_type, misc,
1134 * label, path, first_lineno,
1135 * type, locals, args, exception_table, body]
1136 */
1137
1138 data = CHECK_ARRAY(data);
1139
1140 magic = CHECK_STRING(rb_ary_entry(data, i++));
1141 version1 = CHECK_INTEGER(rb_ary_entry(data, i++));
1142 version2 = CHECK_INTEGER(rb_ary_entry(data, i++));
1143 format_type = CHECK_INTEGER(rb_ary_entry(data, i++));
1144 misc = CHECK_HASH(rb_ary_entry(data, i++));
1145 ((void)magic, (void)version1, (void)version2, (void)format_type);
1146
1147 name = CHECK_STRING(rb_ary_entry(data, i++));
1148 path = CHECK_STRING(rb_ary_entry(data, i++));
1149 realpath = rb_ary_entry(data, i++);
1150 realpath = NIL_P(realpath) ? Qnil : CHECK_STRING(realpath);
1151 first_lineno = CHECK_INTEGER(rb_ary_entry(data, i++));
1152
1153 type = CHECK_SYMBOL(rb_ary_entry(data, i++));
1154 locals = CHECK_ARRAY(rb_ary_entry(data, i++));
1155 params = CHECK_HASH(rb_ary_entry(data, i++));
1156 exception = CHECK_ARRAY(rb_ary_entry(data, i++));
1157 body = CHECK_ARRAY(rb_ary_entry(data, i++));
1158
1159 ISEQ_BODY(iseq)->local_iseq = iseq;
1160
1161 iseq_type = iseq_type_from_sym(type);
1162 if (iseq_type == (enum iseq_type)-1) {
1163 rb_raise(rb_eTypeError, "unsupported type: :%"PRIsVALUE, rb_sym2str(type));
1164 }
1165
1166 node_id = rb_hash_aref(misc, ID2SYM(rb_intern("node_id")));
1167
1168 code_location = rb_hash_aref(misc, ID2SYM(rb_intern("code_location")));
1169 if (RB_TYPE_P(code_location, T_ARRAY) && RARRAY_LEN(code_location) == 4) {
1170 tmp_loc.beg_pos.lineno = NUM2INT(rb_ary_entry(code_location, 0));
1171 tmp_loc.beg_pos.column = NUM2INT(rb_ary_entry(code_location, 1));
1172 tmp_loc.end_pos.lineno = NUM2INT(rb_ary_entry(code_location, 2));
1173 tmp_loc.end_pos.column = NUM2INT(rb_ary_entry(code_location, 3));
1174 }
1175
1176 make_compile_option(&option, opt);
1177 option.peephole_optimization = FALSE; /* because peephole optimization can modify original iseq */
1178 prepare_iseq_build(iseq, name, path, realpath, first_lineno, &tmp_loc, NUM2INT(node_id),
1179 parent, 0, (enum iseq_type)iseq_type, Qnil, &option);
1180
1181 rb_iseq_build_from_ary(iseq, misc, locals, params, exception, body);
1182
1183 finish_iseq_build(iseq);
1184
1185 return iseqw_new(iseq);
1186}

1238 }

1242 rb_exc_raise(GET_EC()->errinfo);

1249 }

1405 * call-seq:
1406 * InstructionSequence.compile(source[, file[, path[, line[, options]]]]) -> iseq
1407 * InstructionSequence.new(source[, file[, path[, line[, options]]]]) -> iseq
1408 *
1409 * Takes +source+, a String of Ruby code and compiles it to an
1410 * InstructionSequence.
1411 *
1412 * Optionally takes +file+, +path+, and +line+ which describe the file path,
1413 * real path and first line number of the ruby code in +source+ which are
1414 * metadata attached to the returned +iseq+.
1415 *
1416 * +file+ is used for `__FILE__` and exception backtrace. +path+ is used for
1417 * +require_relative+ base. It is recommended these should be the same full
1418 * path.
1419 *
1420 * +options+, which can be +true+, +false+ or a +Hash+, is used to
1421 * modify the default behavior of the Ruby iseq compiler.
1422 *
1423 * For details regarding valid compile options see ::compile_option=.
1424 *
1425 * RubyVM::InstructionSequence.compile("a = 1 + 2")
1426 * #=> <RubyVM::InstructionSequence:<compiled>@<compiled>>
1427 *
1428 * path = "test.rb"
1429 * RubyVM::InstructionSequence.compile(File.read(path), path, File.expand_path(path))
1430 * #=> <RubyVM::InstructionSequence:<compiled>@test.rb:1>
1431 *
1432 * path = File.expand_path("test.rb")
1433 * RubyVM::InstructionSequence.compile(File.read(path), path, path)
1434 * #=> <RubyVM::InstructionSequence:<compiled>@/absolute/path/to/test.rb:1>
1435 *
1436 */
1437static VALUE
1438iseqw_s_compile(int argc, VALUE *argv, VALUE self)
1439{
1440 VALUE src, file = Qnil, path = Qnil, line = INT2FIX(1), opt = Qnil;
1441 int i;
1442
1443 i = rb_scan_args(argc, argv, "1*:", &src, NULL, &opt);
1444 if (i > 4+NIL_P(opt)) rb_error_arity(argc, 1, 5);
1445 switch (i) {
1446 case 5: opt = argv[--i];
1447 case 4: line = argv[--i];
1448 case 3: path = argv[--i];
1449 case 2: file = argv[--i];
1450 }
1451
1452 if (NIL_P(file)) file = rb_fstring_lit("<compiled>");
1453 if (NIL_P(path)) path = file;
1454 if (NIL_P(line)) line = INT2FIX(1);
1455
1456 Check_Type(path, T_STRING);
1457 Check_Type(file, T_STRING);
1458
1459 return iseqw_new(rb_iseq_compile_with_option(src, file, path, line, opt));
1460}

1462/*
1463 * call-seq:
1464 * InstructionSequence.compile_file(file[, options]) -> iseq
1465 *
1466 * Takes +file+, a String with the location of a Ruby source file, reads,
1467 * parses and compiles the file, and returns +iseq+, the compiled
1468 * InstructionSequence with source location metadata set.
1469 *
1470 * Optionally takes +options+, which can be +true+, +false+ or a +Hash+, to
1471 * modify the default behavior of the Ruby iseq compiler.
1472 *
1473 * For details regarding valid compile options see ::compile_option=.
1474 *
1475 * # /tmp/hello.rb
1476 * puts "Hello, world!"
1477 *
1478 * # elsewhere
1479 * RubyVM::InstructionSequence.compile_file("/tmp/hello.rb")
1480 * #=> <RubyVM::InstructionSequence:<main>@/tmp/hello.rb>
1481 */
1482static VALUE
1483iseqw_s_compile_file(int argc, VALUE *argv, VALUE self)
1484{
1485 VALUE file, line = INT2FIX(1), opt = Qnil;
1486 VALUE parser, f, exc = Qnil, ret;
1487 rb_ast_t *ast;
1488 rb_compile_option_t option;
1489 int i;
1490
1491 i = rb_scan_args(argc, argv, "1*:", &file, NULL, &opt);
1492 if (i > 1+NIL_P(opt)) rb_error_arity(argc, 1, 2);
1493 switch (i) {
1494 case 2: opt = argv[--i];
1495 }
1496 FilePathValue(file);
1497 file = rb_fstring(file); /* rb_io_t->pathv gets frozen anyways */
1498
1499 f = rb_file_open_str(file, "r");
1500
1501 parser = rb_parser_new();
1502 rb_parser_set_context(parser, NULL, FALSE);
1503 ast = (rb_ast_t *)rb_parser_load_file(parser, file);
1504 if (!ast->body.root) exc = GET_EC()->errinfo;
1505
1506 rb_io_close(f);
1507 if (!ast->body.root) {
1508 rb_ast_dispose(ast);
1509 rb_exc_raise(exc);
1510 }
1511
1512 make_compile_option(&option, opt);
1513
1514 ret = iseqw_new(rb_iseq_new_with_opt(&ast->body, rb_fstring_lit("<main>"),
1515 file,
1516 rb_realpath_internal(Qnil, file, 1),
1517 line, NULL, 0, ISEQ_TYPE_TOP, &option));
1518 rb_ast_dispose(ast);
1519 return ret;
1520}

1850 size_t size = body->insns_info.size;

1865 return &insns_info[0];

1883 }
1884 if (positions[l] > pos) {
1885 return &insns_info[l-1];
1886 }
1887 return &insns_info[l];
1888 }

2174 }

2183 case TS_IVC:
2184 case TS_ICVARC:

2250 const struct rb_builtin_function *bf = (const struct rb_builtin_function *)op;

2252 bf->name, bf->argc);

2569 const struct rb_iseq_constant_body *const body = ISEQ_BODY(iseq);
2570 const rb_iseq_t *child;

2580 rb_hash_aset(all_children, (VALUE)child, Qtrue);

2582 }
2583 }
2584 }
2585 }

2601 }
2602 }
2603 break;
2604 default:
2605 break;
2606 }
2607 }
2608 i += len;
2609 }

3740/* An implementation of succinct bit-vector for insn_info table.
3741 *
3742 * A succinct bit-vector is a small and efficient data structure that provides
3743 * a bit-vector augmented with an index for O(1) rank operation:
3744 *
3745 * rank(bv, n): the number of 1's within a range from index 0 to index n
3746 *
3747 * This can be used to lookup insn_info table from PC.
3748 * For example, consider the following iseq and insn_info_table:
3749 *
3750 * iseq insn_info_table
3751 * PC insn+operand position lineno event
3752 * 0: insn1 0: 1 [Li]
3753 * 2: insn2 2: 2 [Li] <= (A)
3754 * 5: insn3 8: 3 [Li] <= (B)
3755 * 8: insn4
3756 *
3757 * In this case, a succinct bit-vector whose indexes 0, 2, 8 is "1" and
3758 * other indexes is "0", i.e., "101000001", is created.
3759 * To lookup the lineno of insn2, calculate rank("10100001", 2) = 2, so
3760 * the line (A) is the entry in question.
3761 * To lookup the lineno of insn4, calculate rank("10100001", 8) = 3, so
3762 * the line (B) is the entry in question.
3763 *
3764 * A naive implementation of succinct bit-vector works really well
3765 * not only for large size but also for small size. However, it has
3766 * tiny overhead for very small size. So, this implementation consist
3767 * of two parts: one part is the "immediate" table that keeps rank result
3768 * as a raw table, and the other part is a normal succinct bit-vector.
3769 */
3770
3771#define IMMEDIATE_TABLE_SIZE 54 /* a multiple of 9, and < 128 */
3772
3773struct succ_index_table {
3774 uint64_t imm_part[IMMEDIATE_TABLE_SIZE / 9];
3775 struct succ_dict_block {
3776 unsigned int rank;
3777 uint64_t small_block_ranks; /* 9 bits * 7 = 63 bits */
3778 uint64_t bits[512/64];
3779 } succ_part[FLEX_ARY_LEN];
3780};
3781
3782#define imm_block_rank_set(v, i, r) (v) |= (uint64_t)(r) << (7 * (i))
3783#define imm_block_rank_get(v, i) (((int)((v) >> ((i) * 7))) & 0x7f)
3784#define small_block_rank_set(v, i, r) (v) |= (uint64_t)(r) << (9 * ((i) - 1))
3785#define small_block_rank_get(v, i) ((i) == 0 ? 0 : (((int)((v) >> (((i) - 1) * 9))) & 0x1ff))
3786
3787static struct succ_index_table *
3788succ_index_table_create(int max_pos, int *data, int size)
3789{
3790 const int imm_size = (max_pos < IMMEDIATE_TABLE_SIZE ? max_pos + 8 : IMMEDIATE_TABLE_SIZE) / 9;
3791 const int succ_size = (max_pos < IMMEDIATE_TABLE_SIZE ? 0 : (max_pos - IMMEDIATE_TABLE_SIZE + 511)) / 512;
3792 struct succ_index_table *sd =
3793 rb_xcalloc_mul_add_mul(
3794 imm_size, sizeof(uint64_t),
3795 succ_size, sizeof(struct succ_dict_block));
3796 int i, j, k, r;
3797
3798 r = 0;
3799 for (j = 0; j < imm_size; j++) {
3800 for (i = 0; i < 9; i++) {
3801 if (r < size && data[r] == j * 9 + i) r++;
3802 imm_block_rank_set(sd->imm_part[j], i, r);
3803 }
3804 }
3805 for (k = 0; k < succ_size; k++) {
3806 struct succ_dict_block *sd_block = &sd->succ_part[k];
3807 int small_rank = 0;
3808 sd_block->rank = r;
3809 for (j = 0; j < 8; j++) {
3810 uint64_t bits = 0;
3811 if (j) small_block_rank_set(sd_block->small_block_ranks, j, small_rank);
3812 for (i = 0; i < 64; i++) {
3813 if (r < size && data[r] == k * 512 + j * 64 + i + IMMEDIATE_TABLE_SIZE) {
3814 bits |= ((uint64_t)1) << i;
3815 r++;
3816 }
3817 }
3818 sd_block->bits[j] = bits;
3819 small_rank += rb_popcount64(bits);
3820 }
3821 }
3822 return sd;
3823}

3825static unsigned int *
3826succ_index_table_invert(int max_pos, struct succ_index_table *sd, int size)
3827{
3828 const int imm_size = (max_pos < IMMEDIATE_TABLE_SIZE ? max_pos + 8 : IMMEDIATE_TABLE_SIZE) / 9;
3829 const int succ_size = (max_pos < IMMEDIATE_TABLE_SIZE ? 0 : (max_pos - IMMEDIATE_TABLE_SIZE + 511)) / 512;
3830 unsigned int *positions = ALLOC_N(unsigned int, size), *p;
3831 int i, j, k, r = -1;
3832 p = positions;
3833 for (j = 0; j < imm_size; j++) {
3834 for (i = 0; i < 9; i++) {
3835 int nr = imm_block_rank_get(sd->imm_part[j], i);
3836 if (r != nr) *p++ = j * 9 + i;
3837 r = nr;
3838 }
3839 }
3840 for (k = 0; k < succ_size; k++) {
3841 for (j = 0; j < 8; j++) {
3842 for (i = 0; i < 64; i++) {
3843 if (sd->succ_part[k].bits[j] & (((uint64_t)1) << i)) {
3844 *p++ = k * 512 + j * 64 + i + IMMEDIATE_TABLE_SIZE;
3845 }
3846 }
3847 }
3848 }
3849 return positions;
3850}

3852static int
3853succ_index_lookup(const struct succ_index_table *sd, int x)
3854{
3855 if (x < IMMEDIATE_TABLE_SIZE) {
3856 const int i = x / 9;
3857 const int j = x % 9;
3858 return imm_block_rank_get(sd->imm_part[i], j);
3859 }
3860 else {
3861 const int block_index = (x - IMMEDIATE_TABLE_SIZE) / 512;
3862 const struct succ_dict_block *block = &sd->succ_part[block_index];
3863 const int block_bit_index = (x - IMMEDIATE_TABLE_SIZE) % 512;
3864 const int small_block_index = block_bit_index / 64;
3865 const int small_block_popcount = small_block_rank_get(block->small_block_ranks, small_block_index);
3866 const int popcnt = rb_popcount64(block->bits[small_block_index] << (63 - block_bit_index % 64));
3867
3868 return block->rank + small_block_popcount + popcnt;
